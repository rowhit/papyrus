31  /**
32   * Test the expansionmodel model :
33   *  - validate the model
34   * 
35   * @author Benoit Maggi
36   */
37  @SuppressWarnings("nls")
38  public class ViewpointModelTest {
39  	
40  	public static final String VIEWPOINT_MODEL_PATH = Activator.PLUGIN_ID+"/resources/configuration/blockDefinitionDiagram.configuration";
41  	
42  	/**
43  	 * Validate the model with the rules defined in the meta-model tooling
44  	 */
45  	@Ignore //FIXME : the Papyrus default viewpoint is invalid
46  	@Test
47  	public void validateViewpointModel() {
48  		URI createPlatformPluginURI = URI.createPlatformPluginURI(VIEWPOINT_MODEL_PATH, true);
49  		Resource resource = new ResourceSetImpl().getResource(createPlatformPluginURI, true);
50  		Diagnostic diagnostic = Diagnostician.INSTANCE.validate(resource.getContents().get(0));
51  		Assert.assertEquals("The blockDefinitionDiagram.configuration model is not valid ", Diagnostic.OK, diagnostic.getSeverity());
52  	}
53  
54  	
55  	/**
56  	 * Check that the stereotypes are correctly referenced 
57  	 * Can be removed once the validation is working
58  	 */
59  	@Test
60  	public void checkStereotype() {
61  		URI createPlatformPluginURI = URI.createPlatformPluginURI(VIEWPOINT_MODEL_PATH, true);
62  		ResourceSetImpl resourceSetImpl = new ResourceSetImpl();
63  		Resource resource = resourceSetImpl.getResource(createPlatformPluginURI, true);
64  		TreeIterator<EObject> allContents = resource.getAllContents();
65  		while (allContents.hasNext()) {
66  			EObject eObject = (EObject) allContents.next();
67  			if (eObject instanceof ModelRule) { // PapyrusConfiguration
68  				ModelRule modelRule = (ModelRule) eObject;
69  				EList<EClass> stereotypes = modelRule.getStereotypes();
70  				for (EClass eClass : stereotypes) {
71  					Assert.assertFalse(eClass.eIsProxy());
72  				}
73  			}
74  		}
75  	}
