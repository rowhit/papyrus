13  package org.eclipse.papyrus.sysml14.validation.rules;
14  
15  import org.eclipse.core.runtime.IStatus;
16  import org.eclipse.emf.common.util.EList;
17  import org.eclipse.emf.validation.AbstractModelConstraint;
18  import org.eclipse.emf.validation.IValidationContext;
19  import org.eclipse.papyrus.sysml14.requirements.DeriveReqt;
20  import org.eclipse.papyrus.sysml14.requirements.Requirement;
21  import org.eclipse.uml2.uml.Abstraction;
22  import org.eclipse.uml2.uml.NamedElement;
23  import org.eclipse.uml2.uml.util.UMLUtil;
24  /**
25   * 
26   * 16.3.2.2 DeriveReqt (SysML 1.4)
27   * Constraints
28   * [1] The supplier must be an element stereotyped by «requirement» or one of «requirement» subtypes.
29   *
30   */
31  // TODO : I reported an issue since the constraint is not "correct" in SysML 1.4 http://issues.omg.org/browse/SYSMLR-240
32  public class DeriveReqtSupplierModelConstraint extends AbstractModelConstraint {
33  
34  	/* (non-Javadoc)
35  	 * @see org.eclipse.emf.validation.AbstractModelConstraint#validate(org.eclipse.emf.validation.IValidationContext)
36  	 */
37  	@Override
38  	public IStatus validate(IValidationContext context) {
39  		DeriveReqt deriveReqt = (DeriveReqt) context.getTarget();
40  		Abstraction abstraction = deriveReqt.getBase_Abstraction();
41  		if (abstraction != null){
42  			EList<NamedElement> suppliers = abstraction.getSuppliers();
43  			if (suppliers != null && !suppliers.isEmpty()){
44  				if(UMLUtil.getStereotypeApplication(suppliers.get(0), Requirement.class)==null){
45  					return context.createFailureStatus(context.getTarget());
46  				}
47  			}
48  		}
49  		return context.createSuccessStatus();
50  	}
51  
52  }
