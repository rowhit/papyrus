42   * <li>{@link org.eclipse.papyrus.sysml14.blocks.Block#getParts <em>Parts</em>}</li>

44   * </ul>
45   * </p>
46   */
47  public class BlockTest {
48  
49  	private Block owningAgregationBlock=null;
50  
51  	private Block owningCompositionBlock=null;
52  	
53  	private Block flowPropertiesBlock=null;
54  
55  	/**
56  	 * @see junit.framework.TestCase#setUp()
57  	 */
58  	@Before
59  	public void setUp()  {
60  		// Prepare Model test
61  		Model model = SysMLResource.createSysMLModel(new ResourceSetImpl());
62  		Assert.assertFalse("the SysML profil must be applied.", model.getAppliedProfiles().isEmpty());
63  
64  		StereotypeApplicationHelper stereotypeApplicationHelper = StereotypeApplicationHelper.getInstance(null);
65  		/*
66  		 * create an association
67  		 */
68  		Class owningAgregationBlockClass1=model.createOwnedClass("Block1", false);
69  		owningAgregationBlock = (Block) stereotypeApplicationHelper.applyStereotype(owningAgregationBlockClass1, BlocksPackage.eINSTANCE.getBlock());
70  
71  		Class ownedBlockClass2=model.createOwnedClass("Block2", false);
72  		stereotypeApplicationHelper.applyStereotype(ownedBlockClass2, BlocksPackage.eINSTANCE.getBlock());
73  
74  		Property p1Inblock1 = owningAgregationBlockClass1.createOwnedAttribute("P1", ownedBlockClass2);
75  
76  		Association associationAgregation = UMLFactory.eINSTANCE.createAssociation();
77  		Property p1InAsso = associationAgregation.createOwnedEnd("P1", owningAgregationBlockClass1);
78  		associationAgregation.getMemberEnds().add(p1Inblock1);
79  		associationAgregation.getMemberEnds().add(p1InAsso);
80  
81  		p1Inblock1.setAggregation(AggregationKind.SHARED_LITERAL);
82  		p1InAsso.setAggregation(AggregationKind.NONE_LITERAL);
83  
84  		associationAgregation.setPackage(model);
85  
86  		/*
87  		 * create a composition
88  		 */
89  		Class owningCompositionBlockClass3 = model.createOwnedClass("Block3", false);
90  		owningCompositionBlock = (Block) stereotypeApplicationHelper.applyStereotype(owningCompositionBlockClass3, BlocksPackage.eINSTANCE.getBlock());
91  
92  		Property p1Inblock3 = owningCompositionBlockClass3.createOwnedAttribute("P1", ownedBlockClass2);
93  
94  		Association compositionAssociation = UMLFactory.eINSTANCE.createAssociation();
95  		Property p1InAsso3 = compositionAssociation.createOwnedEnd("P1", owningCompositionBlockClass3);
96  		compositionAssociation.getMemberEnds().add(p1Inblock3);
97  		compositionAssociation.getMemberEnds().add(p1InAsso3);
98  
99  		p1Inblock3.setAggregation(AggregationKind.COMPOSITE_LITERAL);
100 		p1InAsso3.setAggregation(AggregationKind.NONE_LITERAL);
101 
102 		compositionAssociation.setPackage(model);
103 		
104 		/*
105 		 * create a flow property and add it to the Block
106 		 */
107 		Class flowPropertyClass = model.createOwnedClass("FPClass1", false);
108 		flowPropertiesBlock = (Block) stereotypeApplicationHelper.applyStereotype(flowPropertyClass, BlocksPackage.eINSTANCE.getBlock());
109 		Property p0 = flowPropertyClass.createOwnedAttribute("p0", null);
110 		StereotypeApplicationHelper.getInstance(null).applyStereotype(p0, PortsandflowsPackage.eINSTANCE.getFlowProperty());
111 	}
112 
113 	/**
114 	 * Tests the '{@link org.eclipse.papyrus.sysml14.blocks.Block#getReferences()
115 	 * <em>Reference</em>}' feature getter. 
116 	 * @see org.eclipse.papyrus.sysml14.blocks.Block#getReferences()
117 	 */
118 	@Test
119 	public void testGetReferences() {  
120 		/*
121 		 * Test if association Agregation is operational
122 		 */
123 		Assert.assertNotNull("References should not be null", owningAgregationBlock.getReferences());
124 		Assert.assertEquals("References size should be 1", 1,owningAgregationBlock.getReferences().size());
125 		/*
126 		 * Test if association Composition is not interfering
127 		 */
128 		Assert.assertNotNull("References should not be null", owningCompositionBlock.getReferences());
129 		Assert.assertEquals("References size should be 0", 0,owningCompositionBlock.getReferences().size());
130 
131 	}
132 
133 	/**
134 	 * Tests the '{@link org.eclipse.papyrus.sysml14.blocks.Block#getParts()
135 	 * <em>Part</em>}' feature getter. 
136 	 * @see org.eclipse.papyrus.sysml14.blocks.Block#getParts()
137 	 */
138 	@Test
139 	public void testGetParts() {  
140 		/*
141 		 * Test if association Composition is not interfering
142 		 */
143 		Assert.assertNotNull("References should not be null", owningCompositionBlock.getParts());
144 		Assert.assertEquals("References size should be 1", 1,owningCompositionBlock.getParts().size());
145 
146 	}
147 
148 
149 	/**
150 	 * Tests the '{@link org.eclipse.papyrus.sysml14.blocks.Block#getFlowProperties()
151 	 * <em>FlowProperty</em>}' feature getter. 
152 	 * @see org.eclipse.papyrus.sysml14.blocks.Block#getFlowProperties()
153 	 */
154 	@Test
155 	public void testGetFlowProperties() {  
156 		Assert.assertNotNull("FlowProperties should not be null", flowPropertiesBlock.getFlowProperties());
157 		Assert.assertEquals("FlowProperties size should be 1", 1, flowPropertiesBlock.getFlowProperties().size());
158 	}
