25  /**
26   * this class has been implemented in order to calculate derived properties of "FlowPort" stereotype see Requirement DerivedProperties Req0010
27   *
28   */
29  public class FlowPortCustomImpl extends FlowPortImpl implements FlowPort {
30  
31  	/**
32  	 * C.3.2.2 FlowPort : /isAtomic : Boolean (derived) This is a derived attribute (derived from the flow portâ€™s type).
33  	 * For a flow port typed by a flow specification the value of this attribute is False, otherwise the value is True.
34  	 * 
35  	 * @see org.eclipse.papyrus.sysml14.deprecatedelements.internal.impl.FlowPortImpl#isAtomic()
36  	 *
37  	 * @return true if atomic
38  	 */
39  	@Override
40  	public boolean isAtomic() {
41  		// The FlowPort is Atomic if it is not typed by a FlowSpecification
42  		boolean isAtomic = true;
43  		if (getBase_Port() != null) {
44  			// Find FlowPort type
45  			Type type = getBase_Port().getType();
46  			if ((type != null) && (type instanceof Interface)) {
47  				FlowSpecification flowSpec = UMLUtil.getStereotypeApplication(type, FlowSpecification.class);
48  				if (flowSpec != null) {
49  					isAtomic = false;
50  				}
51  			}
52  		}
53  		return isAtomic;
54  	}
55  
56  
57  	/**
58  	 * Images registered in Profile are :
59  	 * 0 - FlowPort IN
60  	 * 1 - FlowPort OUT
61  	 * 2 - FlowPort INOUT
62  	 * 3 - FlowPort NA (Non Atomic)
63  	 * 4 (Other) - FlowPort (default, similar to INOUT)
64  	 * @see org.eclipse.papyrus.sysml14.deprecatedelements.internal.impl.FlowPortImpl#getIcon()
65  	 *
66  	 * @return the icon depending on the orientation of the FlowPort
67  	 */
68  	@Override
69  	public Image getIcon() {
70  		Image image = null;
71  		if (getBase_Port() != null) {
72  			Stereotype st = UMLUtil.getStereotype(this);
73  			EList<Image> icons = st.getIcons();
74  			switch (getDirection()) {
75  			case IN:
76  				image = icons.get(FlowDirection.IN.getValue());
77  				break;
78  			case OUT:
79  				image = icons.get(FlowDirection.OUT.getValue());
80  				break;
81  			case INOUT:
82  				image = icons.get(FlowDirection.INOUT.getValue());
83  				break;
84  			default:
85  				image = icons.get(FlowDirection.INOUT.getValue());
86  				break;
87  			}
88  		}
89  		return image;
90  	}
