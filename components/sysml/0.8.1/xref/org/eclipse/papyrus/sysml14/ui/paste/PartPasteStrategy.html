145 	/*
146 	 * (non-Javadoc)
147 	 *
148 	 * @see org.eclipse.papyrus.infra.gmfdiag.common.strategy.paste.IPasteStrategy#getSemanticCommand(org.eclipse.emf.edit.domain.EditingDomain,
149 	 * org.eclipse.emf.ecore.EObject, org.eclipse.papyrus.infra.core.clipboard.PapyrusClipboard)
150 	 */
151 	@Override
152 	public org.eclipse.emf.common.command.Command getSemanticCommand(EditingDomain domain, EObject targetOwner, PapyrusClipboard<Object> papyrusClipboard) {
153 		if (targetOwner instanceof Classifier || targetOwner instanceof Package) { // test if block or package
154 			// get affiliate AdditionnalData
155 			Map<Object, ?> additionalDataMap = papyrusClipboard.getAdditionalDataForStrategy(getID());
156 			if (additionalDataMap != null) {
157 				CompoundCommand compoundCommand = new CompoundCommand("Copy associations for parts"); //$NON-NLS-1$
158 				for (Iterator<Object> iterator = papyrusClipboard.iterator(); iterator.hasNext();) {
159 					Object object = iterator.next();
160 					EObject target = papyrusClipboard.getTragetCopyFromInternalClipboardCopy(object);
161 					Object additionnalData = additionalDataMap.get(object);
162 					EList<Association> listDuplicatedAssociation = new BasicEList<Association>();
163 					EObject associationContainer = null;
164 					if (target instanceof Property && targetOwner instanceof Classifier && additionnalData instanceof PartAdditionalData) { // test if it is a part
165 						Classifier block = (Classifier) targetOwner;
166 						Property property = (Property) target;
167 						PartAdditionalData partAdditionalData = (PartAdditionalData) additionnalData;
168 						Association newAssociation = partAdditionalData.getDuplicatedAssociation();
169 						restoreAssociationPartContext(block, property, newAssociation);
170 						listDuplicatedAssociation.add(newAssociation);
171 						associationContainer = block.eContainer();
172 					} else if (target instanceof Classifier && targetOwner instanceof Package && additionnalData instanceof PartBlockAdditionalData) { // test if it is a block
173 						Classifier classifier = (Classifier) target;
174 						EList<Property> allAttributes = classifier.getAllAttributes();
175 						PartBlockAdditionalData partBlockAdditionalData = (PartBlockAdditionalData) additionnalData;
176 						for (Property property : allAttributes) {
177 							Association duplicatedAssociation = partBlockAdditionalData.getDuplicatedAssociationByPropertyName(property.getName());
178 							if (duplicatedAssociation != null) {
179 								restoreAssociationPartContext(classifier, property, duplicatedAssociation);
180 								listDuplicatedAssociation.add(duplicatedAssociation);
181 							}
182 						}
183 						associationContainer = targetOwner;
184 					}
185 					if (associationContainer != null && !listDuplicatedAssociation.isEmpty()) {
186 						// add associations to the nearest container
187 						MoveRequest moveRequest = new MoveRequest(associationContainer, listDuplicatedAssociation);
188 						IElementEditService provider = ElementEditServiceUtils.getCommandProvider(targetOwner);
189 						if (provider != null) {
190 							ICommand editCommand = provider.getEditCommand(moveRequest);
191 							GMFtoEMFCommandWrapper gmFtoEMFCommandWrapper = new GMFtoEMFCommandWrapper(editCommand);
192 							compoundCommand.append(gmFtoEMFCommandWrapper);
193 						}
194 					}
195 				}
196 				// An empty command can't be executed
197 				if (compoundCommand.getCommandList().isEmpty()) {
198 					return null;
199 				}
200 				return compoundCommand;
201 			}
202 		}
203 		return null;
204 	}

241 	/*
242 	 * (non-Javadoc)
243 	 *
244 	 * @see org.eclipse.papyrus.infra.gmfdiag.common.strategy.paste.IPasteStrategy#prepare(org.eclipse.papyrus.infra.core.clipboard.PapyrusClipboard)
245 	 */
246 	@Override
247 	public void prepare(PapyrusClipboard<Object> papyrusClipboard, Collection<EObject> selection) {
248 		Map<Object, IClipboardAdditionalData> mapCopyPartData = new HashMap<Object, IClipboardAdditionalData>();
249 		for (Iterator<EObject> iterator = papyrusClipboard.iterateOnSource(); iterator.hasNext();) {
250 			EObject eObjectSource = iterator.next();
251 			if (eObjectSource instanceof Element) {
252 				Element elementSource = (Element) eObjectSource;
253 				IClipboardAdditionalData clipboardAdditionalData = null;
254 				if (elementSource instanceof Property && UMLUtil.getStereotypeApplication(elementSource.getOwner(), Block.class) != null) { // is part
255 					Property propertySource = (Property) elementSource;
256 					Association association = propertySource.getAssociation();
257 					if (association != null) {
258 						clipboardAdditionalData = new PartAdditionalData(association);
259 					}
260 				} else if (elementSource instanceof Classifier && UMLUtil.getStereotypeApplication(elementSource, Block.class) != null) {// is Block
261 					Classifier block = (Classifier) elementSource;
262 					clipboardAdditionalData = new PartBlockAdditionalData(block);
263 				}
264 				if (clipboardAdditionalData != null) {
265 					Object copy = papyrusClipboard.getCopyFromSource(eObjectSource);
266 					mapCopyPartData.put(copy, clipboardAdditionalData);
267 				}
268 			}
269 
270 		}
271 		papyrusClipboard.pushAdditionalData(getID(), mapCopyPartData);
272 	}

275 	/**
276 	 * Duplicate the association
277 	 *
278 	 * @param association
279 	 * @return
280 	 */
281 	protected Association duplicateAssociation(Association association) {
282 		EcoreUtil.Copier copier = new EcoreUtil.Copier();
283 		copier.copy(association);
284 		return (Association) copier.get(association);
285 	}
286 
287 	/**
288 	 * Init the association in the new context
289 	 *
290 	 * @param classifier
291 	 * @param property
292 	 * @param association
293 	 */
294 	protected void restoreAssociationPartContext(Classifier classifier, Property property, Association association) {
295 		EList<Property> memberEnds = association.getMemberEnds(); // should have only one element
296 		if (memberEnds != null ) {
297 			if (!memberEnds.isEmpty()){
298 				Property blockProperty = memberEnds.get(0);
299 				blockProperty.setType(classifier);				
300 			}
301 			memberEnds.add(property);
302 		}	
303 	}

307 	 * The Class PartAdditionalData.
308 	 */
309 	protected class PartAdditionalData implements IClipboardAdditionalData {
310 
311 		/** The association. */
312 		protected Association asssociation;
313 
314 		/**
315 		 * @param association
316 		 */
317 		public PartAdditionalData(Association asssociation) {
318 			this.asssociation = duplicateAssociation(asssociation);
319 		}
320 
321 		/**
322 		 * @return a copy of the association
323 		 */
324 		public Association getDuplicatedAssociation() {
325 			return duplicateAssociation(asssociation);
326 		}
327 
328 	}
329 
330 	/**
331 	 * The Class PartBlockAdditionalData.
332 	 */
333 	protected class PartBlockAdditionalData implements IClipboardAdditionalData {
334 
335 		/** The associations by property name. */
336 		protected Map<String, Association> mapPropertyNameToAssociation;
337 
338 		/**
339 		 * @param block
340 		 */
341 		public PartBlockAdditionalData(Classifier block) {
342 			this.mapPropertyNameToAssociation = new HashMap<String, Association>();
343 			EList<Property> allAttributes = block.getAllAttributes();
344 			for (Property property : allAttributes) {
345 				Association association = property.getAssociation();
346 				Association newAssociation = duplicateAssociation(association);
347 				mapPropertyNameToAssociation.put(property.getName(), newAssociation);
348 			}
349 
350 		}
351 
352 		/**
353 		 * @param propertyName
354 		 * @return a copy of the association
355 		 */
356 		public Association getDuplicatedAssociationByPropertyName(String propertyName) {
357 			Association association = mapPropertyNameToAssociation.get(propertyName);
358 			return duplicateAssociation(association);
359 		}
