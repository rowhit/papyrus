12  package org.eclipse.papyrus.sysml14.service.types.advice;
13  
14  import org.eclipse.gmf.runtime.common.core.command.ICommand;
15  import org.eclipse.gmf.runtime.emf.type.core.edithelper.AbstractEditHelperAdvice;
16  import org.eclipse.gmf.runtime.emf.type.core.requests.ConfigureRequest;
17  
18  /**
19   * The EditHelperAdvice class used for:
20   * The property value can imply to launch a popup
21   * this class satisfies the requirement{@papyrus.req org.eclipse.papyrus.sysml14.service.types#req_010_002}
22   *
23   * The reference can imply to launch a popup
24   * this class satisfies the requirement{@papyrus.req org.eclipse.papyrus.sysml14.service.types#req_010_003}
25   *
26   * The part can imply to launch a popup
27   * this class satisfies the requirement{@papyrus.req org.eclipse.papyrus.sysml14.service.types#req_010_004}
28   *
29   * The flowproperty value can imply to launch a popup
30   * this class satisfies the requirement{@papyrus.req org.eclipse.papyrus.sysml14.service.types#req_010_005}
31   */
32  public class AfterConfigureCommandEditHelperAdvice extends AbstractEditHelperAdvice {
33  
34  
35  	/**
36  	 * Parameter id used to pass the command for setting the type of the property
37  	 */
38  	public static final String AFTER_CONFIGURE_COMMAND = "AFTER_CONFIGURE_COMMAND"; //$NON-NLS-1$
39  
40  
41  	/**
42  	 * @see org.eclipse.gmf.runtime.emf.type.core.edithelper.AbstractEditHelperAdvice#getAfterConfigureCommand(org.eclipse.gmf.runtime.emf.type.core.requests.ConfigureRequest)
43  	 *
44  	 * @param request
45  	 * @return
46  	 */
47  	@Override
48  	protected ICommand getAfterConfigureCommand(ConfigureRequest request) {
49  		Object parameter = request.getParameter(AFTER_CONFIGURE_COMMAND);
50  
51  		if (parameter instanceof ICommand) {
52  			return (ICommand) parameter;
53  		}
54  		return super.getAfterConfigureCommand(request);
55  	}
56  }
