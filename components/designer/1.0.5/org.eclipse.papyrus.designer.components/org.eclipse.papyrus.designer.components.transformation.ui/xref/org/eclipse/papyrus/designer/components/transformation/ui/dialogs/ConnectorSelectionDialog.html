81  	public ConnectorSelectionDialog(Shell parent, Package model, Feature selectedConnector) {

92  	public Control createDialogArea(Composite parent) {
93  		Composite contents = (Composite) super.createDialogArea(parent);
94  		createMessageArea(contents);
95  		// createLabel(contents, "Connector type");
96  		GridData gridData = DialogUtils.createFillGridData();
97  		gridData.heightHint = 200;
98  
99  		Group fUpperGroup = new Group(contents, SWT.NONE);
100 		fUpperGroup.setLayout(new GridLayout());
101 		fUpperGroup.setLayoutData(gridData);
102 		fUpperGroup.setText(" Abstract interaction components "); //$NON-NLS-1$
103 		createFilterText(fUpperGroup);
104 		fUpper = createFilteredList(fUpperGroup);
105 
106 		final Group fLowerGroup = new Group(contents, SWT.NONE);
107 		fLowerGroup.setLayout(new GridLayout());
108 		fLowerGroup.setLayoutData(gridData);
109 		fLowerGroup.setText(" Realizations/groups of interaction components "); //$NON-NLS-1$
110 		// createLabel(contents, lowerLabel);
111 		fLower = createFilteredList(fLowerGroup);
112 		fFilteredList = fUpper;
113 		org.eclipse.papyrus.designer.components.FCM.Connector fcmConnector = UMLUtil.getStereotypeApplication(m_selectedConnector, org.eclipse.papyrus.designer.components.FCM.Connector.class);
114 		if (fcmConnector != null) {
115 			// getInitialElementSelections();
116 			initialSelection = fcmConnector.getIc();
117 		} else {
118 			initialSelection = null;
119 		}
120 
121 		fTypeOnly = new Button(contents, SWT.CHECK);
122 		fTypeOnly.setText("Select type only"); //$NON-NLS-1$
123 		SelectionListener fTypeOnlySL = new SelectionListener() {

125 			@Override

133 				else {

135 					fLowerGroup.setEnabled(true);
136 					handleSelectionChanged();
137 				}
138 			}
139 
140 			@Override
141 			public void widgetDefaultSelected(SelectionEvent e) {
142 			}
143 		};
144 		fTypeOnly.addSelectionListener(fTypeOnlySL);
145 
146 		fUpper.setElements(connectorList.toArray());
147

149 		if (initialSelection instanceof InteractionComponent) {
150 			InteractionComponent componentComp = initialSelection;

152 			if (!componentCompBase.isAbstract()) {

154 				Class componentTypeBase = ElementUtils.componentType(componentCompBase);

157 				// fLower.setEnabled (false);

159 				fTypeOnly.setSelection(true);
160 			}
161 			// update status (listener does not evaluate passed data)
162 			fTypeOnlySL.widgetSelected(null);
163 
164 			// set initial selections
165 			fUpper.setSelection(new Object[] { componentComp.getBase_Class() });
166 		}
167 
168 		Label info = createMessageArea(contents);
169 		info.setText("Information about connector implementation:"); //$NON-NLS-1$

171 
172 		GridData descriptionGD = new GridData();
173 		descriptionGD.horizontalAlignment = GridData.FILL;
174 		descriptionGD.grabExcessHorizontalSpace = true;
175 		descriptionGD.heightHint = 80;
176 		descriptionGD.grabExcessVerticalSpace = true;
177 		descriptionGD.verticalAlignment = GridData.FILL;
178 
179 		m_description = new Text(contents, SWT.NONE | SWT.WRAP | SWT.V_SCROLL | SWT.READ_ONLY);
180 		m_description.setLayoutData(descriptionGD);
181 
182 		fLower.addSelectionListener(new SelectionListener() {
183 
184 			@Override

186 				Object[] selected = fLower.getSelection();

193 					updateOkState();

271 	private void handleUpperSelectionChanged() {
272 		int index = getSelectionIndex();
273 		if (index >= 0) {
274 			fUpperElements = getFoldedElements(index);
275 
276 			if ((fUpperElements == null) || fTypeOnly.getSelection()) {
277 				// lower list remains empty, if typeOnly is true
278 				fLower.setElements(new Object[] {});
279 			} else {
280 				fLower.setElements(getLowerList(fUpperElements[0]));
281 				if (initialSelection instanceof InteractionComponent) {
282 					// select existing selection in lower section
283 					fLower.setSelection(new Object[] { initialSelection });
284 					// do not try to set it again.
285 					initialSelection = null;
286 				}
287 			}
288 		}
289 		validateCurrentSelection();
290 	}

292 	/**
293 	 * Returns the selected element from the upper or lower pane.
294 	 *
295 	 * @return Object
296 	 */
297 	protected Object getSelectedElement(boolean upper) {
298 		Object[] selection;
299 		if (upper) {
300 			selection = fFilteredList.getSelection();
301 		} else {
302 			selection = fLower.getSelection();
303 		}
304 		if (selection.length > 0) {
305 			return selection[0];
306 		}
307 		return null;
308 	}

310 	protected Object[] getLowerList(Object selectedUpperObj) {
311 		EList<Class> connectorList = new UniqueEList<Class>();
312 		if (selectedUpperObj instanceof Class) {
313 			visitedPackages = new BasicEList<Package>();
314 			getImplGroups(m_model, (Class) selectedUpperObj, connectorList);
315 		}
316 		return connectorList.toArray();
317 	}

319 	void getAllConnectors(Package pkg, Feature selectedConnector, EList<Class> connectorList) {
320 		for (Element el : pkg.getMembers()) {
321 			if (el instanceof Package) {
322 				if (!visitedPackages.contains(el)) {
323 					visitedPackages.add((Package) el);
324 					getAllConnectors((Package) el, selectedConnector, connectorList);
325 				}
326 			} else if (el instanceof Class) {
327 				if (StereotypeUtil.isApplied(el, InteractionComponent.class) && ((Class) el).isAbstract()) {
328 
329 					if (selectedConnector == null) {
330 						connectorList.add((Class) el);
331 					} else {
332 						try {
333 							Element owner = selectedConnector.getOwner();
334 							if (owner instanceof Class) {
335 								Class composite = (Class) owner;
336 								Activator.log.info("ConnectorSelectionDialog.getAllConnectors: try to bind connector " + //$NON-NLS-1$
337 										((Class) el).getQualifiedName());
338 								ConnectorBinding.obtainBinding(composite, selectedConnector, (Class) el, false);
339 								connectorList.add((Class) el);
340 							} else {
341 								connectorList.add((Class) el);
342 							}
343 						} catch (TransformationException e) {
344 							// silently ignore exception: it is normal that we cannot find a binding for some connectors
345 						}
346 					}
347 				}
348 			}
349 		}
350 	}
351 
352 	void getImplGroups(Package pkg, Class selectedConnType, EList<Class> implGroupList) {
353 		for (Element el : pkg.getMembers()) {
354 			if (el instanceof Package) {
355 				if (!visitedPackages.contains(el)) {
356 					visitedPackages.add((Package) el);
357 					getImplGroups((Package) el, selectedConnType, implGroupList);
358 				}
359 			} else if (el instanceof Class) {
360 				Class class_ = (Class) el;
361 				if (class_.getGeneralization(selectedConnType) != null) {
362 					implGroupList.add((Class) el);
363 				}
364 			}
365 		}
