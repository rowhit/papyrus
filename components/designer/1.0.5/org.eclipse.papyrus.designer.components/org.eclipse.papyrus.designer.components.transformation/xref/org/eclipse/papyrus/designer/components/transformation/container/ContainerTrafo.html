98  	/**
99  	 * creates the executor. Needs to be called *before* the other operations of
100 	 * this class.
101 	 *
102 	 * @param tmComponent
103 	 *            the implementation of a component
104 	 *
105 	 * @throws TransformationException
106 	 */
107 	@Override
108 	public void createContainer(Class smComponent, Class tmComponent) throws TransformationException {
109 		Package tmPkgOwner;
110 		if (tmComponent.eResource() == tmCDP.eResource()) {
111 			// tmComponent is in same model as tmContainer
112 			tmPkgOwner = tmComponent.getNearestPackage();
113 		}
114 		else {
115 			// tmComponent is not a member of the target model. This is the case, if tmComponent is
116 			// in an imported model that has not been copied during the transformation towards an intermediate
117 			// model.
118 			tmPkgOwner = MapUtil.getAndCreate(PackageUtil.getRootPackage(tmCDP), tmComponent.allNamespaces(), true);
119 		}
120 		// create a container with the suitable postfix
121 		tmContainerImpl = tmPkgOwner.createOwnedClass(tmComponent.getName() + containerPostfix, false);
122 
123 		// Apply singleton to container, if original class is a singleton
124 		boolean isSingleton = StereotypeUtil.isApplied(smComponent, Singleton.class);
125 		if (isSingleton) {
126 			StereotypeUtil.apply(tmContainerImpl, Singleton.class);
127 		}
128 		CopyUtils.copyID(tmComponent, tmContainerImpl, containerPostfix);
129 
130 		// add part and slot corresponding to component;
131 		executorPart = tmContainerImpl.createOwnedAttribute(executorPartName, tmComponent);
132 		CopyUtils.copyID(tmComponent, executorPart, "e"); //$NON-NLS-1$
133 
134 		executorPart.setIsComposite(true);
135 
136 		// copy ports from the executor (tmComponent) to the container
137 		// (tmContainerImpl)
138 		// TODO: connectors still point to the ports of the executor, implying a non-well-formed
139 		// model. Yet code generation works, as port names are identical
140 		for (Port port : PortUtils.getAllPorts(tmComponent)) {
141 			// copy port
142 			if (port.getVisibility() == VisibilityKind.PUBLIC_LITERAL) {
143 				Port newPort = EcoreUtil.copy(port); // don't use copier, since this is not a copier from source to target model
144 				tmContainerImpl.getOwnedAttributes().add(newPort);
145 				StUtils.copyStereotypes(port, newPort);
146 
147 				// create delegation for application port
148 				Connector containerDelegation = tmContainerImpl.createOwnedConnector("delegation " + port.getName()); //$NON-NLS-1$
149 				CopyUtils.copyID(tmContainerImpl, containerDelegation);
150 				ConnectorEnd end1 = containerDelegation.createEnd();
151 				end1.setRole(newPort);
152 				ConnectorEnd end2 = containerDelegation.createEnd();
153 				end2.setPartWithPort(executorPart);
154 				end2.setRole(port);
155 			}
156 		}
157 
158 		// Inheritance issues:
159 		// currently: container never copies inheritance relationships, but copies inherited ports into container
160 		// Problem: polymorphism would stop working.
161 		// Objective: use inheritance, if possible, but avoid that container inherits from an original non-abstract class (avoid duplication of attributes, ...)
162 		// Option: container inherits from fully abstract super-classes (might have to consider indirect super-classes).
163 		// All connectors and container extension remain within container (i.e. no inheritance of connectors and container extensions)
164 		// Remaining problems: Imagine that component A inherits from component B and both have state-charts.
165 		// => containers of A and B would define a state-Machine and an event pool.
166 		// => No shared pool (but anyway: semantics of this construct is not very clear).
167 		containers.put(tmComponent, this);
168 	}

180 		Package tmPkgOwner = (Package) tmComponent.getOwner();
181 		tmContainerImpl = tmPkgOwner.createOwnedClass(tmComponent.getName() + hwContainerPostfix, false);
182 		CopyUtils.copyID(tmComponent, tmContainerImpl, hwContainerPostfix);

185 		containers.put(tmComponent, this);

188 	/**
189 	 * Create an instance of the container (UML instance specification)
190 	 *
191 	 * @param tmComponent
192 	 *            the component within the target model for which we want to create an instance
193 	 *
194 	 * @param tmIS
195 	 *            the existing instance specification for the component before container expansion. This
196 	 *            instance specification becomes the new instance specification of the executor.
197 	 * @param context
198 	 *            Additional information about the container that is used by instance configurators
199 	 */
200 	@Override
201 	public InstanceSpecification createContainerInstance(Class tmComponent, InstanceSpecification tmIS) throws TransformationException {
202 		// create an instance specification for the container
203 		InstanceSpecification containerIS =
204 				(InstanceSpecification) tmCDP.createPackagedElement(tmIS.getName(), UMLPackage.eINSTANCE.getInstanceSpecification());
205 		// assign new name to original instance specification which reflects
206 		// role within containment hierarchy
207 		tmIS.setName(tmIS.getName() + "." + executorPartName); //$NON-NLS-1$
208 		containerIS.getClassifiers().add(tmContainerImpl);
209 		// existing instance specification (tmIS) is the executor instance specification
210 		DepCreation.createSlot(containerIS, tmIS, executorPart);
211 		counter = 0;
212 		// containers.put(tmComponent, this);
213 		EList<Slot> connectorSlots = new BasicEList<Slot>();
214 
215 		executorIS = tmIS;
216 		// now create instances for the contained elements
217 		for (Property extensionPart : tmContainerImpl.getAttributes()) {
218 			Type tmContainerExtImpl = extensionPart.getType();
219 			if (tmContainerExtImpl instanceof Class) {
220 				if (DepUtils.getSlot(containerIS, extensionPart) == null) {
221 					// no slot for part exists => assume that the part has been added by the container and create an instance specification for it.
222 					InstanceSpecification containerExtIS = null;
223 
224 					String isName = containerIS.getName() + "." + extensionPart.getName(); //$NON-NLS-1$
225 
226 					// create sub-instance and slot for extensions
227 					if (containerExtIS == null) {
228 						containerExtIS = DepCreation.createDepPlan(tmCDP, (Class) tmContainerExtImpl, isName, false);
229 					}
230 					Slot partSlot = DepCreation.createSlot(containerIS, containerExtIS, extensionPart);
231 					if (StereotypeUtil.isApplied(tmContainerExtImpl, InteractionComponent.class)) {
232 						connectorSlots.add(partSlot);
233 					}
234 				}
235 			}
236 		}
237 
238 		// handle propagation of node allocation for connectors. Do that in 2nd loop, since the allocation
239 		// of parts might be incomplete before.
240 		// TODO: is in library now
241 		//for (Slot connectorSlot : connectorSlots) {
242 		//	ConnectorReification.propagateNodeAllocation(tmContainerImpl, containerIS, connectorSlot);
243 		// }
244 		moveSlots(executorIS);
245 		return containerIS;
246 	}

285 	/**
286 	 * apply a container rule, i.e. add either a container extension or an
287 	 * interceptor to the container
288 	 *
289 	 * @param smContainerRule
290 	 *            An container rule
291 	 * @param smComponent
292 	 *            the application component in the source model
293 	 * @param tmComponent
294 	 *            the application component in the target model
295 	 * @param tmIS
296 	 *            the instance specification for the application component in the target model
297 	 * @throws TransformationException
298 	 */
299 	@Override
300 	public void applyRule(ContainerRule smContainerRule, Class smComponent, Class tmComponent)
301 			throws TransformationException
302 	{
303 		Map<Property, EList<Property>> interceptorPartsMap = new HashMap<Property, EList<Property>>();
304 
305 		for (Property part : smContainerRule.getBase_Class().getAllAttributes()) {
306 			Type type = part.getType();
307 			if (type == null) {
308 				String ruleName = (smContainerRule.getBase_Class() != null) ? smContainerRule.getBase_Class().getName() : "undefined"; //$NON-NLS-1$
309 				throw new TransformationException(String.format(
310 						Messages.ContainerTrafo_CannotApplyRule, ruleName));
311 			}
312 			if (part instanceof Port) {
313 				Port newPort = tmContainerImpl.createOwnedPort(part.getName(), part.getType());
314 				StUtils.copyStereotypes(part, newPort);
315 			}
316 			else if (type instanceof Class) {
317 				Class extOrInterceptor = (Class) type;
318 				// DepUtils.chooseImplementation((Class) type,
319 				// new BasicEList<InstanceSpecification>(), false);
320 
321 				if (StereotypeUtil.isApplied(part, InterceptionRule.class)) {
322 					if (StereotypeUtil.isApplied(extOrInterceptor, InteractionComponent.class)) {
323 
324 						InterceptionRule interceptionRule = UMLUtil.getStereotypeApplication(part, InterceptionRule.class);
325 						InterceptionKind interceptionKind = interceptionRule.getInterceptionKind();
326 						EList<Feature> interceptFeatures = interceptionRule.getInterceptionSet();
327 						EList<Property> interceptorParts =
328 								expandInterceptorExtension(interceptionKind, interceptFeatures, extOrInterceptor, tmComponent);
329 						interceptorPartsMap.put(part, interceptorParts);
330 					}
331 					else {
332 						throw new TransformationException(String.format(
333 								Messages.ContainerTrafo_InterceptionRuleButNoInterceptor,
334 								part.getName(), smContainerRule.getBase_Class().getName()));
335 					}
336 				}
337 				else {
338 					Property extensionPart =
339 							expandAggregationExtension(part, extOrInterceptor, tmComponent);
340 					// register relation to facilitate connector copy
341 					copier.setPackageTemplate(smContainerRule.getBase_Class(), tmContainerImpl);
342 					copier.putPair(part, extensionPart);
343 					copier.setPackageTemplate(null, null);
344 				}
345 			}
346 		}
347 		createConnectorForAssociations();
348 
349 		// tell copier that tmcontainerImpl is associated with the smContainerRule
350 		// register a package template (although it is not a template) to assure that the connectors
351 		// get copied, although they are in a different resource (only the connectors are copied, not
352 		// the types of the referenced parts).
353 		// [main issue here: properties of container rule are not copies in the sense of identical
354 		// copies]
355 		copier.setPackageTemplate(smContainerRule.getBase_Class(), tmContainerImpl);
356 
357 		for (Connector connector : smContainerRule.getBase_Class().getOwnedConnectors()) {
358 			Property ruleInterceptorPart = null;
359 			for (Property part : interceptorPartsMap.keySet()) {
360 				if (ConnectorUtil.connectsPart(connector, part)) {
361 					// this connector is a connection between an interceptor (and another part which may not
362 					// be an interceptor => TODO: validation rule yet to write
363 					// we need to copy this connector multiple times, once for each associated interceptor part
364 					ruleInterceptorPart = part;
365 					break;
366 				}
367 			}
368 			if (ruleInterceptorPart != null) {
369 				for (Property interceptorPart : interceptorPartsMap.get(ruleInterceptorPart)) {
370 					// map an interceptor part within rule successively to to an interceptor part
371 					// within the container
372 					copier.putPair(ruleInterceptorPart, interceptorPart);
373 					copier.getCopy(connector);
374 				}
375 			}
376 			else {
377 				// check whether FCM connector
378 				org.eclipse.papyrus.designer.components.FCM.Connector fcmConn =
379 						UMLUtil.getStereotypeApplication(connector, org.eclipse.papyrus.designer.components.FCM.Connector.class);
380 				if (fcmConn != null) {
381 					//
382 					// TODO: is in library now
383 					// ConnectorReification.reifyConnector(copier, tmContainerImpl,
384 					//		ElementUtils.varName(connector), connector, executorIS);
385 				}
386 				else {
387 					copier.remove(connector);
388 					copier.getCopy(connector);
389 				}
390 			}
391 		}
392 		copier.setPackageTemplate(null, null);
393 
394 		TemplateUtils.retargetConnectors(tmContainerImpl);
395 	}

397 	/**
398 	 * This container expansion does not create a new composite. Instead, it
399 	 * adds the extension as a part to the copied application component. The
400 	 * application component also inherits from the type of the container
401 	 * extension in order to avoid copying ports.
402 	 */
403 	Property expandAggregationExtension(Property smExtensionPart, Class smContainerExtImpl, Class tmComponent) throws TransformationException {
404 		Property extensionPart;
405 		String name = smExtensionPart.getName();
406 		Class tmContainerExtImpl = null;
407 
408 		TemplateSignature signature = TemplateUtils.getSignature(smContainerExtImpl);
409 		if (signature == null) {
410 			// no template signature, just copy the container extension into the target model
411 			tmContainerExtImpl = copier.getCopy(smContainerExtImpl);
412 		}
413 		else {
414 			// template signature found, instantiate container extension via template binding mechanism
415 			// use executor component as actual
416 			Classifier actual = tmComponent;
417 			// template signature and instantiate container extension via the
418 			// template binding mechanism, use executor component as actual
419 			TemplateBinding binding = TemplateUtils.fixedBinding(copier.target, smContainerExtImpl, actual);
420 			TemplateInstantiation ti = new TemplateInstantiation(copier, binding);
421 			tmContainerExtImpl = ti.bindElement(smContainerExtImpl);
422 		}
423 
424 		// --------------------------------------------------
425 		// a container extension could be subject to container rules as well, i.e. need
426 		// to handle recursive rule application.
427 		EList<ContainerRule> rules = FCMUtil.getAllContainerRules(smContainerExtImpl);
428 
429 		// get container trafo instance, if already existing
430 		AbstractContainerTrafo containerTrafo = AbstractContainerTrafo.get(tmContainerExtImpl);
431 
432 		if (containerTrafo == null) {
433 			// no container exists, check rules and create eventually
434 			for (ContainerRule rule : rules) {
435 				// at least one active rule => create container (or get previously instantiated))
436 				if (containerTrafo == null) {
437 					if (rule.getKind() == ContainerRuleKind.LIGHT_WEIGHT_OO_RULE) {
438 						throw new TransformationException(Messages.ContainerTrafo_RecursiveLWnotSupported);
439 					}
440 					else {
441 						containerTrafo = new ContainerTrafo(copier, tmCDP, null);
442 					}
443 					containerTrafo.createContainer(smContainerExtImpl, tmContainerExtImpl);
444 				}
445 				else {
446 					// configure only??
447 				}
448 				containerTrafo.applyRule(rule, smContainerExtImpl, tmContainerExtImpl);
449 			}
450 			if (containerTrafo != null) {
451 				containerTrafo.finalize();
452 				tmContainerExtImpl = ((ContainerTrafo) containerTrafo).getContainer();
453 			}
454 		}
455 		// --------------------------------------------------
456 
457 		// add part associated with the extension to the container
458 		extensionPart = tmContainerImpl.createOwnedAttribute(name, tmContainerExtImpl);
459 
460 		// Copy.copyID(tmComponent, extensionPart, "a");
461 		extensionPart.setAggregation(smExtensionPart.getAggregation());
462 		CopyUtils.copyMultElemModifiers(smExtensionPart, extensionPart);
463 		CopyUtils.copyFeatureModifiers(smExtensionPart, extensionPart);
464 
465 		return extensionPart;
466 	}

468 	/**
469 	 * Expand an interceptor definition between the container and the executor.
470 	 *
471 	 * @param extKind
472 	 *            the interception kind (intercept which ports: all, some, out, ...)
473 	 * @param featureList
474 	 *            The set of ports to intercept (In case of "some" above)
475 	 * @param smContainerConnImpl
476 	 *            The connector (interceptor) implementation in the source model
477 	 * @param tmComponent
478 	 *            the application component (executor) in the target model
479 	 * @return
480 	 * @throws TransformationException
481 	 */
482 	EList<Property> expandInterceptorExtension(InterceptionKind extKind, EList<Feature> featureList, Class smContainerConnImpl, Class tmComponent)
483 			throws TransformationException
484 	{
485 		EList<Property> connectorParts = new BasicEList<Property>();
486 		// replace delegation connectors with extension connector
487 		for (Port port : PortUtils.getAllPorts(tmComponent)) {
488 			// delegate to component via connector
489 			boolean match = true;
490 			if (extKind == InterceptionKind.INTERCEPT_ALL_IN) {
491 				// IN-PORT = provided port
492 				match = (PortUtils.getProvided(port) != null);
493 			}
494 			else if (extKind == InterceptionKind.INTERCEPT_ALL_OUT) {
495 				// IN-PORT = provided port
496 				match = (PortUtils.getRequired(port) != null);
497 			}
498 			else if (extKind == InterceptionKind.INTERCEPT_SOME) {
499 				// comparison based on name, since in different models
500 				match = (ElementUtils.getNamedElementFromList(featureList, port.getName()) != null);
501 			}
502 			else if (extKind == InterceptionKind.INTERCEPT_MATCHING) {
503 				EList<Port> interceptorPorts = PortUtils.getAllPorts(smContainerConnImpl);
504 				match = false;
505 				// get first port from interception connector that is typed with a template parameter
506 				for (Port interceptorPort : interceptorPorts) {
507 					if (interceptorPort.getType().getOwner() instanceof TemplateParameter) {
508 						match = (PortUtils.getKind(port) == PortUtils.getKind(interceptorPort));
509 						break;
510 					}
511 				}
512 			}
513 
514 			// else INTERCEPT_ALL_PORT => match remains true
515 			if (!match) {
516 				// port does not match criterion, continue with next port
517 				continue;
518 			}
519 
520 			Property connectorPart;
521 			String interceptionName = interceptorName + port.getName() + counter;
522 			Connector interceptionConnector = null;
523 			// get delegation connector
524 			for (Connector connector : tmContainerImpl.getOwnedConnectors()) {
525 				if (ConnectorUtil.connectsPort(connector, port)) {
526 					interceptionConnector = connector;
527 					break;
528 				}
529 			}
530 			// interceptionConnector = tmContainerImpl.getOwnedConnector
531 			// ("delegation " + port.getName ());
532 			if (interceptionConnector == null) {
533 				throw new TransformationException(Messages.ContainerTrafo_CannotFindDelegationConn);
534 			}
535 
536 			interceptionConnector.setName(interceptionName);
537 			org.eclipse.papyrus.designer.components.FCM.Connector fcmConn = StereotypeUtil.applyApp(interceptionConnector, org.eclipse.papyrus.designer.components.FCM.Connector.class);
538 			InteractionComponent fcmConnType = UMLUtil.getStereotypeApplication(smContainerConnImpl, InteractionComponent.class);
539 			fcmConn.setIc(fcmConnType);
540 
541 			// pass target component and port to interceptor (not clean, define
542 			// suitable template signature as for instance in methodCall_comp
543 			ContainerTrafo.instance = executorIS;
544 			ContainerTrafo.port = port;
545 			// TODO: is in library now
546 			// connectorPart = ConnectorReification.reifyConnector(copier, tmContainerImpl, ElementUtils.varName(interceptionConnector), interceptionConnector, executorIS);
547 			connectorPart = null;
548 			connectorParts.add(connectorPart);
549 			ContainerTrafo.port = null;
550 			portInfo.put(connectorPart, port);
551 			// delete intermediate connector (has been replaced by two
552 			// connections to the reified connector)
553 			interceptionConnector.destroy();
554 		}
555 		counter++;
556 		return connectorParts;
557 	}

559 	/**
560 	 * Move a slot from the executor instance specification to an instance
561 	 * specification of an extension/interceptor. This is required, since
562 	 * users cannot configure containers directly. The modification is done in
563 	 * the target model, i.e. the user model is not affected.
564 	 *
565 	 * Configuration is based on the idea that we can have multiple classifiers for an
566 	 * instance specification and that we add additional slots for attributes of the 2nd
567 	 * classifier which becomes "visible" after container transformation.
568 	 * Main use: configure priority of thread within container.
569 	 *
570 	 * TODO: Not clear, whether we should keep this operation, since configuration of elements that are not
571 	 * (directly) part of the user model is confusing. Application was: configuration
572 	 * of thread priorities within container.
573 	 */
574 	public void moveSlots(InstanceSpecification containerIS) {
575 		// Log.log(Status.INFO, Log.TRAFO_CONTAINER, String.format(Messages.ContainerTrafo_InfoMoveSlots, executorIS.getQualifiedName()));
576 		Classifier mainCl = DepUtils.getClassifier(executorIS);
577 		Iterator<Slot> slotIt = executorIS.getSlots().iterator();
578 		while (slotIt.hasNext()) {
579 			Slot slot = slotIt.next();
580 			String featureName = slot.getDefiningFeature().getName();
581 			if (!mainCl.getAllAttributes().contains(slot.getDefiningFeature())) {
582 				// defining feature does not belong to main classifier
583 				// => move it to first matching instance specification
584 				for (InstanceSpecification is : DepUtils.getContainedInstances(containerIS)) {
585 					Classifier containedCl = DepUtils.getClassifier(is);
586 					if (ElementUtils.getNamedElementFromList(containedCl.getAllAttributes(), featureName) != null) {
587 						if (executorIS != is) {
588 							// remove slot first from iterator, as addition below removes it from the list (slots are owned)
589 							slotIt.remove();
590 							is.getSlots().add(slot);
591 							break;
592 						}
593 					}
594 				}
595 			}
596 		}
597 	}

599 	/**
600 	 * Create an connector for parts that have an association. The rationale
601 	 * behind this is support connections between the state machine in the container
602 	 * and the executor.
603 	 *
604 	 */
605 	public void createConnectorForAssociations() {
606 		// TODO: keep list of added parts, only re-check those!
607 
608 		for (Property part : PartsUtil.getParts(tmContainerImpl)) {
609 			if (part.getType() == null) {
610 				continue;
611 			}
612 			for (Association association : part.getType().getAssociations()) {
613 				// Part has an association. Check if the other association end
614 				// targets one of
615 				// other parts within the container
616 				for (Property end : association.getMemberEnds()) {
617 					Type type = end.getType();
618 					if ((type != null) && (type != part.getType())) {
619 						// end is not typed with the part we are currently
620 						// examining
621 						// thus, it might belong to another part of the
622 						// composite
623 						for (Property checkPart : PartsUtil.getParts(tmContainerImpl)) {
624 							if (type == checkPart.getType()) {
625 								// found an association between two parts of the
626 								// container => create connection,
627 								// unless already existing.
628 								if (!ConnectorUtil.existsConnector(tmContainerImpl, part, checkPart)) {
629 									Connector conn = tmContainerImpl.createOwnedConnector(part.getName() + "_" + checkPart.getName()); //$NON-NLS-1$
630 									conn.setType(association);
631 									conn.createEnd().setRole(part);
632 									conn.createEnd().setRole(checkPart);
633 								}
634 							}
635 						}
636 					}
637 				}
638 			}
639 		}
640 	}

647 	/**
648 	 * Connector ports of executor with compatible ports offered by container extensions
649 	 * (runtime system)
650 	 */
651 	public void discoverServices() {
652 		Type executorType = executorPart.getType();
653 		if (!(executorType instanceof EncapsulatedClassifier)) {
654 			return;
655 		}
656 		EncapsulatedClassifier ec = (EncapsulatedClassifier) executorType;
657 		for (Port executorPort : ec.getOwnedPorts()) {
658 			for (Property svcPart : tmContainerImpl.getOwnedAttributes()) {
659 				if (svcPart == executorPart) {
660 					continue;
661 				}
662 				Type containerSvcType = svcPart.getType();
663 				if (containerSvcType instanceof EncapsulatedClassifier) {
664 					EncapsulatedClassifier containerSvc = (EncapsulatedClassifier) containerSvcType;
665 					for (Port svcPort : containerSvc.getOwnedPorts()) {
666 						if (PortUtils.matches(executorPort, svcPort, true)) {
667 							// create connector
668 							Connector c = tmContainerImpl.createOwnedConnector(
669 									String.format("auto from %s to %s", executorPart.getName(), svcPart.getName())); //$NON-NLS-1$
670 							ConnectorEnd ce1 = c.createEnd();
671 							ConnectorEnd ce2 = c.createEnd();
672 							ce1.setPartWithPort(executorPart);
673 							ce1.setRole(executorPort);
674 							ce2.setPartWithPort(svcPart);
675 							ce2.setRole(svcPort);
676 						}
677 					}
678 				}
679 			}
680 		}
681 	}
