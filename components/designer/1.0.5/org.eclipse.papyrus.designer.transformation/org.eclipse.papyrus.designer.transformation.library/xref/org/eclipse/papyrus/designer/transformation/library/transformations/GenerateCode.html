33  	public void generate(InstanceSpecification node, String targetLanguage)
34  			throws TransformationException, InterruptedException
35  	{
36  		IProgressMonitor monitor = UIContext.monitor;
37  		IProject genProject = TransformationContext.current.project;
38  		
39  		Package genModel = TransformationContext.current.mm.getModel();
40  
41  		monitor.worked(1);
42  		if (genProject == null) {
43  			return;
44  		}
45  
46  		if (node == null) {
47  			monitor.setTaskName(String.format(Messages.GenerateCode_GeneratingCode, targetLanguage));
48  		}
49  		else {
50  			monitor.setTaskName(String.format(Messages.GenerateCode_GeneratingCodeForNode, targetLanguage, node.getName()));
51  		}
52  		IFolder folder = genProject.getFolder(GenUtils.getSourceFolder(genModel));
53  		try {
54  			folder.delete(true, null);
55  		} catch (CoreException e) {
56  			throw new TransformationException(String.format(Messages.GenerateCode_CouldNotDeleteOldCode, e.getMessage()));
57  		}
58  		ILangCodegen codegen = LanguageCodegen.getGenerator(targetLanguage);
59  	
60  		codegen.generateCode(genProject, genModel, monitor);
61  
62  		// the generated model can contain more than one top-level element due to copied external model references
63  		for (ModelManagement mm : TransformationContext.current.copier.getAdditionalRootPkgs()) {
64  			codegen.generateCode(genProject, mm.getModel(), monitor);
65  		}
66  
67  		if (monitor.isCanceled()) {
68  			return;
69  		}
70  		monitor.worked(1);
71  	}

73  	/**
74  	 * @param element
75  	 * @return
76  	 */
77  	public PackageableElement getNearestClassifierOrPackage(Element element) {
78  		while (element != null) {
79  			if (element instanceof Classifier) {
80  				return (Classifier) element;
81  			}
82  			if (element instanceof Package) {
83  				return (Package) element;
84  			}
85  			element = element.getOwner();
86  		}
87  		return null;
88  	}

128 	@Override
129 	public void applyTrafo(M2MTrafo trafo, Package deploymentPlan) throws TransformationException {
130 		// get first language (restricted to single target language, acceptable?)
131 
132 		String targetLanguage = DepUtils.getTargetLanguage(DepUtils.getTopLevelInstances(deploymentPlan).iterator().next());
133 		try {
134 			generate(null, targetLanguage);
135 		}
136 		catch (InterruptedException e) {
137 			Activator.log.error(e);
138 		}
139 	}
