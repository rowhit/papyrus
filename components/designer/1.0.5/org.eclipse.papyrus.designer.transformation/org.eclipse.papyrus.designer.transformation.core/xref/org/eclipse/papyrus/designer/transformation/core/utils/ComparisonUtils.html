15  package org.eclipse.papyrus.designer.transformation.core.utils;
16  
17  import org.eclipse.emf.ecore.EObject;
18  import org.eclipse.papyrus.designer.transformation.core.transformations.LazyCopier;
19  import org.eclipse.papyrus.designer.transformation.core.transformations.TransformationContext;
20  
21  /**
22   * Check whether two references identify the "same" element, but take into account that one of these
23   * might already have been copied. 
24   *
25   */
26  public class ComparisonUtils {
27  	/**
28  	 * @param a an eObject
29  	 * @param b another eObject
30  	 * @return true, if the objects are identical or if one of these has been copied and the copy
31  	 * corresponds to the same object 
32  	 */
33  	public static boolean sameObject(EObject a, EObject b) {
34  		LazyCopier copier = (TransformationContext.current != null) ? TransformationContext.current.copier : null;
35  		if (copier != null) {
36  			return (a == b) || (copier.get(a) == b) || (copier.get(b) == a);
37  		}
38  		else {
39  			return a == b;
40  		}
41  	}
42  }
