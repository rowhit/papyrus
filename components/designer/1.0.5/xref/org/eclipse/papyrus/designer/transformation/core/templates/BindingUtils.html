29  public class BindingUtils {
30  
31  	/**
32  	 * Bind an operation by copying the signature from the actual.
33  	 *
34  	 * @param copier
35  	 *            the copier
36  	 * @param actual
37  	 *            the actual. If an operation, its signature is copied to the template
38  	 * @param operation
39  	 *            The operation template
40  	 * @return
41  	 */
42  	public static void instantiateOperation(Element actual, Operation operation) {
43  		try {
44  			TransformationContext.current.classifier = operation.getClass_();
45  			// set new name
46  			String newName = TextTemplateBinding.bind(operation.getName(), actual, null);
47  			operation.setName(newName);
48  			// perform binding in case of C++ initializer
49  			ConstInit cppConstInit = UMLUtil.getStereotypeApplication(operation, ConstInit.class);
50  			if (cppConstInit != null) {
51  				// TODO: specific to C++
52  				String init = cppConstInit.getInitialisation();
53  				String newInit = TextTemplateBinding.bind(init, actual);
54  				cppConstInit.setInitialisation(newInit);
55  			}
56  		} catch (TransformationException e) {
57  			// throw runtime exception
58  			throw new RuntimeException(String.format(Messages.TemplateInstantiationListener_TrafoException, e.getMessage()));
59  		}
60  	}
61  
62  	/**
63  	 * Instantiate a behavior, i.e. check whether the body text consists of a text template
64  	 * reference.  If yes, instantiate the text template.
65  	 *
66  	 * @param copier
67  	 *            copier
68  	 * @param actual
69  	 *            actual in template instantiation
70  	 * @param opaqueBehavior
71  	 *            behavior with body containing eventually a text template reference
72  	 * @return instantiated (bound) behavior.
73  	 * @throws TransformationException
74  	 */
75  	public static void instantiateBehavior(Element actual, OpaqueBehavior opaqueBehavior) throws TransformationException {
76  		if (actual instanceof NamedElement) {
77  			String newName = TextTemplateBinding.bind(opaqueBehavior.getName(), actual, null);
78  			if (newName != null && !newName.equals(opaqueBehavior.getName())) {
79  				opaqueBehavior.setName(newName);
80  			}
81  		}
82  		EList<String> bodyList = opaqueBehavior.getBodies();
83  		for (int i = 0; i < bodyList.size(); i++) {
84  			String body = bodyList.get(i);
85  			TransformationContext.current.classifier = (Classifier) opaqueBehavior.getOwner();
86  			// pass qualified operation name as template name. Used to identify script in case of an error
87  			String newBody = TextTemplateBinding.bind(body, actual);
88  			if (newBody != null && !newBody.equals(body)) {
89  				bodyList.set(i, newBody);
90  			}
91  		}
92  	}
93  }
