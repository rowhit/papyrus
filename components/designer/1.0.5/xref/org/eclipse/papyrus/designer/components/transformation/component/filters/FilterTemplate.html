27  /**
28   * Do not copy connector related elements, i.e.
29   * 1. UML connectors with stereotype FCM.Connector
30   * The rationale behind this is that we want to avoid copying a FCM connector from the source model
31   * into the target model since the target model should only contain the reified form of the connector
32   *
33   * 2. UML classes that are stereotyped as interaction components
34   * We do not want to copy interaction components, since they typically need to be instantiated.
35   * TODO: there is no check whether we are actually in a template. An interaction component might be fixed (i.e. not
36   * depend on template instantiation) and needs to be copied in this case.
37   * (in most cases, these are not copied anyway, since these are in a different model,
38   * but we do not want to exclude the case of defining model-local connectors)
39   */
40  public class FilterTemplate implements PreCopyListener {
41  
42  	public FilterTemplate() {
43  		active = true;
44  	}
45  
46  	public static FilterTemplate getInstance() {
47  		if (instance == null) {
48  			instance = new FilterTemplate();
49  		}
50  		return instance;
51  	}
52  
53  	@Override
54  	public EObject preCopyEObject(LazyCopier copier, EObject sourceEObj) {
55  		if (active && (sourceEObj instanceof Element)) {
56  			Element sourceElem = (Element) sourceEObj;
57  			if ((sourceEObj instanceof Connector) && StereotypeUtil.isApplied(sourceElem, org.eclipse.papyrus.designer.components.FCM.Connector.class)) {
58  	//			return null;
59  			}
60  			if ((sourceEObj instanceof Class) && StereotypeUtil.isApplied(sourceElem, InteractionComponent.class)) {
61  				return null;
62  			}
63  		}
64  		return sourceEObj;
65  	}
66  
67  	public void setActive(boolean active) {
68  		this.active = active;
69  	}
70  
71  	private boolean active;
72  
73  	private static FilterTemplate instance = null;
74  }
