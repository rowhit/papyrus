28  /**
29   * An extended Port in the sense of the DDS4CCM standard: a port typed with a component
30   * type (the extended port, not the data type that is transported). Since the component typing
31   * the port can have multiple provided and required
32   * ports, the ports are implicitly owned by the port. Conjugation on the level of an extended
33   * port level will conjugate all ports of the component typing the port.
34   *
35   * The derived interface that is provided will allow a caller to access individual ports. It
36   * is therefore a reference to the interfaces get_<portName> operations provided by a
37   * component.
38   * The derived interface that is required is identical, except for a conjugation on the
39   * extended port level.
40   *
41   * What is the difference to ExtendedPort?
42   *
43   */
44  public class ExtendedPort2 implements IMappingRule {
45  
46  	public static final String CONJ_PREFIX = "C2_"; //$NON-NLS-1$
47  
48  	public static final String NORM_PREFIX = "N2_"; //$NON-NLS-1$
49  
50  	@Override
51  	public Type calcDerivedType(org.eclipse.papyrus.designer.components.FCM.Port p, boolean update) {
52  	
53  		Type type = p.getType();
54  		if (!(type instanceof Class)) {
55  			return null;
56  		}
57  
58  		Class extendedPort = (Class) type;
59  		boolean isConjugated = p.getBase_Port().isConjugated();
60  		String prefix = isConjugated ? CONJ_PREFIX : NORM_PREFIX;
61  		Class derivedClass = PortMapUtil.getDerivedClass(p, prefix, update);
62  		Interface providedInterface = PortMapUtil.getDerivedInterface(p, prefix, update);
63  		Interface requiredInterface = PortMapUtil.getDerivedInterface(p, prefix, update);
64  		if (!update) {
65  			return derivedClass;
66  		}
67  		if (derivedClass == null) {
68  			return null;
69  		}
70  		for (Port port : extendedPort.getOwnedPorts()) {
71  			// if the extended port is conjugated, each of the provided/required are (implicitly)
72  			// conjugated [TODO: is PortUtils aware of it? - probably yes]
73  			Interface reqIntf = PortUtils.getRequired(port);
74  		
75  			if (reqIntf != null) {
76  				String name = PrefixConstants.getP_Prefix + port.getName();
77  
78  				// check whether operation already exists. Create, if not
79  				Operation derivedOperation = requiredInterface.getOperation(name, null, null);
80  				if (derivedOperation == null) {
81  					derivedOperation = requiredInterface.createOwnedOperation(name, null, null);
82  				}
83  			}
84  			
85  			Interface provIntf = PortUtils.getProvided(port);
86  			
87  			if (provIntf != null) {
88  				String name = PrefixConstants.getConnQ_Prefix + port.getName();
89  
90  				// check whether operation already exists. Create, if not
91  				Operation derivedOperation = providedInterface.getOperation(name, null, null);
92  				if (derivedOperation == null) {
93  					derivedOperation = providedInterface.createOwnedOperation(name, null, null);
94  				}
95  			}
96  		}
97  		return derivedClass;
98  	}
99  
100 	@Override
101 	public boolean needsUpdate(org.eclipse.papyrus.designer.components.FCM.Port p) {
102 		return needsUpdate(p, false) ||
103 				needsUpdate(p, true);
104 	}
105 
106 	public boolean needsUpdate(org.eclipse.papyrus.designer.components.FCM.Port p, boolean isConjugated) {
107 		Type type = p.getBase_Port().getType();
108 		if (!(type instanceof Class)) {
109 			return false;
110 		}
111 
112 		Class extendedPort = (Class) type;
113 		String prefix = isConjugated ? CONJ_PREFIX : NORM_PREFIX;
114 		Interface derivedInterface = PortMapUtil.getDerivedInterface(p, prefix);
115 
116 		if (derivedInterface == null) {
117 			return true;
118 		}
119 		for (Port port : extendedPort.getOwnedPorts()) {
120 			// if the extended port is conjugated, each of the provided/required are (implicitly)
121 			// conjugated [TODO: is PortUtils aware of it? - probably yes]
122 			Interface provIntf = (isConjugated) ?
123 					PortUtils.getRequired(port) :
124 					PortUtils.getProvided(port);
125 
126 			if (provIntf != null) {
127 				String name = PrefixConstants.getP_Prefix + port.getName();
128 
129 				// check whether operation already exists. Create, if not
130 				Operation derivedOperation = derivedInterface.getOperation(name, null, null);
131 				if (derivedOperation == null) {
132 					return true;
133 				}
134 			}
135 		}
136 		return false;
137 	}
