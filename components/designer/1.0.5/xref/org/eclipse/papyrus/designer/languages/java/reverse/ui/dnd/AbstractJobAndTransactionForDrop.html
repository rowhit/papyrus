14  package org.eclipse.papyrus.designer.languages.java.reverse.ui.dnd;
15  
16  import java.util.LinkedList;
17  import java.util.List;
18  
19  import org.eclipse.emf.ecore.EObject;
20  import org.eclipse.gef.EditPart;
21  import org.eclipse.gef.Request;
22  import org.eclipse.gef.requests.ChangeBoundsRequest;
23  import org.eclipse.gef.requests.GroupRequest;
24  import org.eclipse.gmf.runtime.diagram.ui.requests.DropObjectsRequest;
25  import org.eclipse.gmf.runtime.notation.View;
26  import org.eclipse.papyrus.designer.languages.java.reverse.ui.exception.StopExecutionException;
27  import org.eclipse.papyrus.infra.emf.utils.EMFHelper;
28  
29  /**
30   * @author dumoulin
31   *
32   */
33  public abstract class AbstractJobAndTransactionForDrop implements IJobAndTransactionForDrop {
34  
35  	
36  	protected Request request;
37  	protected EditPart targetEditPart;
38  	/**
39  	 * Constructor.
40  	 *
41  	 */
42  	public AbstractJobAndTransactionForDrop() {
43  		// TODO Auto-generated constructor stub
44  	}
45  
46  	/**
47  	 * @see org.eclipse.papyrus.designer.languages.java.reverse.ui.dnd.IJobAndTransactionForDrop#init(org.eclipse.gef.Request, org.eclipse.gef.EditPart)
48  	 *
49  	 * @param request
50  	 * @param targetEditPart
51  	 * @throws StopExecutionException
52  	 */
53  	@Override
54  	public void init(Request request, EditPart targetEditPart) throws StopExecutionException {
55  		this.request = request;
56  		this.targetEditPart = targetEditPart;
57  
58  	}
59  	
60  	/**
61  	 * Returns the list of selected objects. The objects can be of any type.
62  	 *
63  	 * @param request
64  	 *            The drop request
65  	 * @return
66  	 *         The list of dropped Objects
67  	 */
68  	protected List<Object> getSourceObjects(Request request) {
69  		List<Object> result = new LinkedList<Object>();
70  		if (request instanceof DropObjectsRequest) {
71  			List objects = ((DropObjectsRequest) request).getObjects();
72  			if (objects != null) {
73  				result.addAll(objects);
74  			}
75  		} else if (request instanceof GroupRequest) {
76  			List editParts = ((ChangeBoundsRequest) request).getEditParts();
77  			if (editParts != null) {
78  				result.addAll(editParts);
79  			}
80  		}
81  		return result;
82  	}
83  
84  	/**
85  	 * Returns the list of semantic EObjects being dropped. If some of the dropped elements
86  	 * cannot be resolved to EObjects, they are ignored.
87  	 *
88  	 * @param request
89  	 *            The drop request
90  	 * @return
91  	 *         The list of dropped EObjects
92  	 */
93  	protected List<EObject> getSourceEObjects(Request request) {
94  		List<EObject> result = new LinkedList<EObject>();
95  
96  		for (Object object : getSourceObjects(request)) {
97  			EObject eObject = EMFHelper.getEObject(object);
98  			if (eObject instanceof View) {
99  				eObject = ((View) eObject).getElement();
100 			}
101 
102 			if (eObject != null) {
103 				result.add(eObject);
104 			}
105 		}
106 
107 		return result;
108 	}
109 
110 }
