14  package org.eclipse.papyrus.designer.ucm.core.commands;
15  
16  import org.eclipse.core.commands.ExecutionException;
17  import org.eclipse.core.commands.operations.IUndoContext;
18  import org.eclipse.core.runtime.IAdaptable;
19  import org.eclipse.core.runtime.IProgressMonitor;
20  import org.eclipse.gmf.runtime.common.core.command.CommandResult;
21  import org.eclipse.gmf.runtime.common.core.command.ICommand;
22  import org.eclipse.gmf.runtime.common.core.command.UnexecutableCommand;
23  import org.eclipse.gmf.runtime.diagram.ui.commands.PopupMenuCommand;
24  import org.eclipse.gmf.runtime.diagram.ui.menus.PopupMenu;
25  import org.eclipse.swt.widgets.Shell;
26  
27  public class PopupMenuCommandHelper {
28  	
29  	public static ICommand getPopupMenuCommand(Shell parentShell, PopupMenu popupMenu) {
30  		return new PopupMenuCommand("Set Type", parentShell, popupMenu) {
31  			private ICommand delegate;
32  
33  			@Override
34  			protected CommandResult doExecuteWithResult(
35  					IProgressMonitor progressMonitor, IAdaptable info)
36  					throws ExecutionException {
37  
38  				if (!getPopupMenu().show(getParentShell())) {
39  					// User cancelled the menu
40  					delegate = UnexecutableCommand.INSTANCE;
41  					progressMonitor.setCanceled(true);
42  					return CommandResult.newCancelledCommandResult();
43  				}
44  
45  				delegate = (ICommand) getPopupMenu().getResult();
46  				delegate.execute(progressMonitor, info);
47  
48  				// Inherit contexts
49  				for (IUndoContext next : delegate.getContexts()) {
50  					if (!hasContext(next)) {
51  						addContext(next);
52  					}
53  				}
54  
55  				return delegate.getCommandResult();
56  			}
57  
58  			@Override
59  			public boolean canUndo() {
60  				return (delegate != null) && delegate.canUndo();
61  			}
62  
63  			@Override
64  			protected CommandResult doUndoWithResult(IProgressMonitor progressMonitor, IAdaptable info) throws ExecutionException {
65  				delegate.undo(progressMonitor, info);
66  				return delegate.getCommandResult();
67  			}
68  
69  			@Override
70  			public boolean canRedo() {
71  				return (delegate != null) && delegate.canRedo();
72  			}
73  
74  			@Override
75  			protected CommandResult doRedoWithResult(IProgressMonitor progressMonitor, IAdaptable info) throws ExecutionException {
76  				delegate.redo(progressMonitor, info);
77  				return delegate.getCommandResult();
78  			}
79  		};
80  	}
81  }
