105 	/**
106 	 * Instantiate a deployment plan, i.e. generate an intermediate model via a
107 	 * sequence of transformations
108 	 *
109 	 * @param umlElement
110 	 *            a deployment plan (UML package) or a configuration (UML class)
111 	 * @param monitor
112 	 *            a progress monitor.
113 	 *
114 	 * @param project
115 	 *            the current project. This information is used to store the
116 	 *            intermediate model in a subfolder (tmpModel) of the current
117 	 *            project
118 	 * @param genOptions
119 	 *            select whether to produce an intermediate model only, also
120 	 *            code, ... @see GenerationOptions
121 	 */
122 	private void instantiate(IProgressMonitor monitor, int genOptions) {
123 		try {
124 			initialize(monitor, genOptions);
125 			executeTransformation();
126 		} catch (final TransformationException e) {
127 			printAndDisplayErrorMessage(e, Messages.InstantiateDepPlan_TransformationException, false);
128 		} catch (final Exception e) {
129 			printAndDisplayErrorMessage(e, Messages.InstantiateDepPlan_ErrorsDuringTransformation, true);
130 		} finally {
131 			TransformationContext.current = null;
132 		}
133 	}

135 	private void initialize(IProgressMonitor monitor, int genOptions) throws TransformationException {
136 		this.monitor = monitor;
137 		this.generationOptions = genOptions;
138 		this.generateCode = (genOptions & GenerationOptions.MODEL_ONLY) == 0;
139 		this.generateCACOnly = (genOptions & GenerationOptions.CAC_ONLY) != 0;
140 	}

142 	protected void executeTransformation() throws Exception {
143 		ModelManagement intermediateModelManagement = null;
144 		UIContext.monitor = monitor;
145 		UIContext.configureProject = (generationOptions & GenerationOptions.REWRITE_SETTINGS) != 0;
146 
147 		// 1a: create a new model (and applies same profiles / imports)
148 		Model existingModel = srcModelComponentDeploymentPlan.getModel();
149 		TransformationContext tc = new TransformationContext();
150 		TransformationContext.current = tc;
151 		TransformationContext.initialSourceRoot = existingModel;
152 		TransformationContext.initialDeploymentPlan = srcModelComponentDeploymentPlan;
153 
154 		intermediateModelManagement = ModelManagement.createNewModel(existingModel.getName());
155 
156 		// get the temporary model
157 		Package intermediateModel = intermediateModelManagement.getModel();
158 
159 		// create a package for global enumerations that are used by xtend code
160 		EnumService.createEnumPackage(intermediateModel);
161 
162 		// create a lazy copier towards the intermediate model
163 		LazyCopier intermediateModelCopier = new LazyCopier(existingModel, intermediateModel, CopyExtResources.SELECTED, true);
164 
165 		// obtain the component deployment plan in target model, only make a shallow
166 		// copy right now, otherwise all instances would already be copied.
167 		intermediateModelCopier.shallowCopy(srcModelComponentDeploymentPlan);
168 
169 		InstanceConfigurator.onNodeModel = false;
170 
171 		tc.copier = intermediateModelCopier;
172 		Map<InstanceSpecification, InstanceSpecification> instanceMap = new HashMap<InstanceSpecification, InstanceSpecification>();
173 
174 		// calculate which resources to copy: based on the classifiers that are referenced from the instances
175 		for (InstanceSpecification instance : DepUtils.getInstances(srcModelComponentDeploymentPlan)) {
176 			if (!TemplateUtils.withinPkgTemplate(instance)) {
177 				Classifier cl = DepUtils.getClassifier(instance);
178 				intermediateModelCopier.addResource(cl.eResource());
179 			}
180 		}
181 
182 		for (InstanceSpecification instance : DepUtils.getTopLevelInstances(srcModelComponentDeploymentPlan)) {
183 			// InstanceSpecification newInstance = mainModelTrafo.transformInstance(instance, null);
184 			InstanceSpecification newInstance = intermediateModelCopier.getCopy(instance);
185 
186 			// --------------------------------------------------------------------
187 			checkProgressStatus();
188 			// --------------------------------------------------------------------
189 
190 			TransformationUtil.applyInstanceConfigurators(newInstance);
191 
192 			TransformationUtil.propagateAllocation(newInstance);
193 			instanceMap.put(instance, newInstance);
194 		}
195 		tc.deploymentPlan = intermediateModelCopier.getCopy(srcModelComponentDeploymentPlan);
196 		tc.modelRoot = PackageUtil.getRootPackage(tc.deploymentPlan);
197 
198 		M2MTrafoChain chain = DepUtils.getTransformationChain(srcModelComponentDeploymentPlan);
199 
200 		ExecuteTransformationChain.apply(chain, DepUtils.getAdditionalTransformations(srcModelComponentDeploymentPlan));
201 
202 		/*
203 		 * if (!generateCACOnly) {
204 		 * deployOnNodes(instanceMap, existingModel, intermediateModel);
205 		 * }
206 		 */
207 
208 		// setURIs and save (first update all URIs to ensure that model references use the
209 		// correct URI before saving)
210 		intermediateModelManagement.setURI(project, TEMP_MODEL_FOLDER, TEMP_MODEL_POSTFIX);
211 		// also save additional root elements
212 		for (ModelManagement mm : tc.copier.getAdditionalRootPkgs()) {
213 			mm.setURI(project, TEMP_MODEL_FOLDER, TEMP_MODEL_POSTFIX);
214 		}
215 		intermediateModelManagement.save();
216 		for (ModelManagement mm : tc.copier.getAdditionalRootPkgs()) {
217 			mm.save();
218 		}
219 
220 		// --------------------------------------------------------------------
221 		checkProgressStatus();
222 		// --------------------------------------------------------------------
223 
224 		intermediateModelManagement.dispose();
225 		// also dispose additional models
226 		for (ModelManagement mm : tc.copier.getAdditionalRootPkgs()) {
227 			mm.dispose();
228 		}
229 	}

238 	private void removeDerivedInterfacesInRoot(Model generatedModel) {
239 		// 2b: remove derived interfaces in root: derived interfaces
240 		// that can not be placed in the same package as the port type (e.g.
241 		// since read-only type from system library), are put in a top-level
242 		// package called "derivedInterfaces". Due to the copying of imports,
243 		// the top-level package has changed which implies that new derived
244 		// interfaces are put into a different package and the derivedInterfaces
245 		// package in the original root becomes obsolete. Delete this obsolete
246 		// package, if existing.
247 		for (PackageableElement packagedElement : generatedModel.getPackagedElements()) {
248 			if (packagedElement instanceof Package) {
249 				NamedElement derivedInterfaces = ((Package) packagedElement).getPackagedElement("derivedInterfaces"); //$NON-NLS-1$
250 				if (derivedInterfaces instanceof Package) {
251 					derivedInterfaces.destroy();
252 				}
253 			}
254 		}
255 	}

258 	 *
259 	 * @param canonicalProjectName
260 	 *            the automatically calculated project name
261 	 * @param userProjectName
262 	 *            the project name choosen by the user
263 	 */
264 	public void updateProjectMapping(final String canonicalProjectName, final String userProjectName) {
265 		CommandSupport.exec(srcModelComponentDeploymentPlan, "Update project mapping", new Runnable() {
266 
267 			@Override
268 			public void run() {
269 				DeploymentPlan depPlan = UMLUtil.getStereotypeApplication(srcModelComponentDeploymentPlan, DeploymentPlan.class);
270 				String mapName = canonicalProjectName + "=" + userProjectName;
271 				for (String mapping : depPlan.getProjectMappings()) {
272 					if (mapping.startsWith(canonicalProjectName)) {
273 						mapping = mapName;
274 						return;
275 					}
276 				}
277 				depPlan.getProjectMappings().add(mapName);
278 			}
279 		});
280 
281 	}

283 	private void checkProgressStatus() throws InterruptedException {
284 		if (monitor.isCanceled()) {
285 			throw new InterruptedException();
286 		}
287 		monitor.worked(1);
288 	}

290 	private void printAndDisplayErrorMessage(Exception e, final String title, final boolean consultConsole) {
291 		String message = e.toString();
292 		if (consultConsole) {
293 			message = message + "\n\n" //$NON-NLS-1$
294 					+ Messages.InstantiateDepPlan_ConsultConsole;
295 		}
296 
297 		printAndDisplayErrorMessage(e, title, message, consultConsole);
298 	}
299 
300 	private void printAndDisplayErrorMessage(Exception e, final String title, final String message, final boolean consultConsole) {
301 		e.printStackTrace();
302 		// only display message, if not running headless
303 		if (!TestInfo.runsHeadless()) {
304 			displayError(title, message);
305 		}
306 		Activator.log.error(e);
307 	}
308 
309 	private void displayError(final String title, final String message) {
310 		Display.getDefault().syncExec(new Runnable() {
311 			@Override
312 			public void run() {
313 				Shell shell = Display.getDefault().getActiveShell();
314 				MessageDialog.openInformation(shell, title, message);
315 			}
316 		});
317 	}
