158 	/**
159 	 * Examine the children of a translation unit in order to extract the
160 	 * methods that are defined within hte unit
161 	 *
162 	 * @param itu
163 	 * @param selector
164 	 * @param parent
165 	 * @throws CModelException
166 	 */
167 	public void examineChildren(ITranslationUnit itu, IASTNodeSelector selector, IParent parent)
168 			throws CModelException {
169 
170 		int position = 0;
171 		// if (parent instanceof Namespace) {
172 		for (ICElement child : parent.getChildren()) {
173 			if (child instanceof IParent) {
174 				examineChildren(itu, selector, (IParent) child);
175 			}
176 			ISourceRange range = null;
177 			if (child instanceof ISourceReference) {
178 				range = ((ISourceReference) child).getSourceRange();
179 			}
180 			if (child instanceof IFunctionDeclaration) {
181 				// function declaration is a superclass for method declaration
182 				// (but need to trace functions differently?)
183 				String name = ((IFunctionDeclaration) child).getElementName();
184 				IASTNode node = selector.findEnclosingNode(range.getStartPos(), range.getLength());
185 				if (node instanceof IASTFunctionDefinition) {
186 					IASTFunctionDefinition definition = (IASTFunctionDefinition) node;
187 					IASTFunctionDeclarator declarator = definition.getDeclarator();
188 					String unfilteredBody = getBody(itu, definition);
189 					// get additional information about method synchronization from generator
190 					SyncInformation syncInfo = null;
191 					if (m_codegen instanceof ILangCodegen2) {
192 						syncInfo = ((ILangCodegen2) m_codegen).getSyncInformation(name, unfilteredBody);
193 					}
194 					String body = Utils.removeGenerated(unfilteredBody);
195 					if (syncInfo == null || !syncInfo.isGenerated) {
196 						// only update method, if it is not generated 
197 						NamedElement ne = updateMethod(position, parent, name, body, declarator, syncInfo);
198 						updateComment(itu, definition, ne);
199 					}
200 					// System.err.println("body source <" + body + ">");
201 				}
202 				position++;
203 			}
204 		}
205 	}

207 	/**
208 	 * update the contents of the CppInclude directive
209 	 *
210 	 * @param itu
211 	 *            the translation unit
212 	 */
213 	public void updateCppInclude(ITranslationUnit itu) {
214 		String contents = new String(itu.getContents());
215 		int preBodyStart = contents.indexOf(Constants.cppIncPreBodyStart);
216 		int preBodyEnd = contents.indexOf(Constants.cppIncPreBodyEnd);
217 		String preBody = ""; //$NON-NLS-1$
218 		String body = ""; //$NON-NLS-1$
219 		if (preBodyStart != -1) {
220 			preBodyStart += Constants.cppIncPreBodyStart.length();
221 			if (preBodyEnd > preBodyStart) {
222 				preBody = contents.substring(preBodyStart, preBodyEnd).trim();
223 			}
224 		}
225 
226 		int bodyStart = contents.indexOf(Constants.cppIncBodyStart);
227 		int bodyEnd = contents.indexOf(Constants.cppIncBodyEnd);
228 
229 		if (bodyStart != -1) {
230 			bodyStart += Constants.cppIncBodyStart.length() + 1;
231 			if (bodyEnd > bodyStart) {
232 				body = contents.substring(bodyStart, bodyEnd).trim();
233 			}
234 		}
235 		if (body.length() > 0 || preBody.length() > 0) {
236 			Include include = StereotypeUtil.applyApp(m_classifier, Include.class);
237 			if (include != null) {
238 				include.setPreBody(preBody);
239 				include.setBody(body);
240 			}
241 		}
242 	}

244 	/**
245 	 * Update a method in the model based on the qualified name.
246 	 *
247 	 * @param position
248 	 *            The position of the method within the file. Used to identify
249 	 *            renaming operations
250 	 * @param parent
251 	 *            the CDT parent which is used to get a list of children
252 	 * @param qualifiedName
253 	 *            the qualified name of a method
254 	 * @param body
255 	 *            the method body
256 	 * @param declarator
257 	 *            the declarator for the method
258 	 * @return the operation or the behavior within the model that got updated. The latter is returned in
259 	 *            case of behaviors that do not have a specification (e.g. the effect of a transition).
260 	 */
261 	public NamedElement updateMethod(int position, IParent parent, String qualifiedName, String body,
262 			IASTFunctionDeclarator declarator, SyncInformation syncInfo) {
263 
264 		String names[] = qualifiedName.split(Utils.nsSep);
265 		String name = names[names.length - 1];
266 
267 		Operation operation = null;
268 		Behavior behavior = null;
269 
270 		if (syncInfo == null || (syncInfo.behavior == null && syncInfo.createBehaviorName == null)) {
271 			operation = getModelOperationFromName(name, parent, position);
272 			if (operation != null) {
273 				operation.setName(name);
274 			} else {
275 				// it is possible that the C++ method corresponds to the effect
276 				// of a transition. try to locate the behavior (without using an operation)
277 				behavior = FindTransition.findBehavior(m_classifier, name);
278 				if (behavior == null) {
279 					// still null => create new operation in model
280 					if (m_classifier instanceof Class) {
281 						operation = ((Class) m_classifier).createOwnedOperation(name, null, null);
282 					} else if (m_classifier instanceof DataType) {
283 						operation = ((DataType) m_classifier).createOwnedOperation(name, null, null);
284 					}
285 				}
286 			}
287 		}
288 		else if (syncInfo.behavior != null) {
289 			// operation is still null (=> does not enter operation != null case below)
290 			behavior = syncInfo.behavior;
291 		}
292 		else if ((syncInfo.createBehaviorName != null) && (m_classifier instanceof Class)) {
293 			Class clazz = (Class) m_classifier;
294 			behavior = (OpaqueBehavior) clazz.createOwnedBehavior(syncInfo.createBehaviorName,
295 					UMLPackage.eINSTANCE.getOpaqueBehavior().eClass());
296 		}
297 
298 		if (operation != null) {
299 			if (operation.getMethods().size() == 0) {
300 				// operation exists, but does not have any method => create
301 
302 				if (m_classifier instanceof Class) {
303 					behavior = ((Class) m_classifier).createOwnedBehavior(name, UMLPackage.eINSTANCE.getOpaqueBehavior());
304 				} else if (m_classifier instanceof DataType) {
305 					// ob = (OpaqueBehavior) ((DataType)
306 					// m_classifier).createOwnedBehavior(name,
307 					// UMLPackage.eINSTANCE.getOpaqueBehavior());
308 				}
309 				behavior.setSpecification(operation);
310 				behavior.setIsReentrant(false);
311 			}
312 			else {
313 				behavior = operation.getMethods().get(0);
314 				// operation has at least one method, this may not be null.
315 				if (!behavior.getName().equals(name)) {
316 					behavior.setName(name);
317 				}
318 			}
319 		}
320 
321 		// assertions: operation can be null, behavior is always non-null
322 		
323 		// Remove all parameters from operation / behavior (they will be added later).
324 		// Calling parameters.clear() is not sufficient. Otherwise stereotype
325 		// applications to unresolved elements remain in the model
326 		EList<Parameter> existingParameters = new BasicEList<Parameter>();
327 		if (operation != null) {
328 			for (Parameter existingParameter : operation.getOwnedParameters()) {
329 				Parameter existingParamCopy = UMLFactory.eINSTANCE.createParameter();
330 				existingParamCopy.setName(existingParameter.getName());
331 				existingParamCopy.setType(existingParameter.getType());
332 				existingParameters.add(existingParamCopy);
333 			}
334 			UMLUtil.destroyElements(operation.getOwnedParameters());
335 		}
336 		UMLUtil.destroyElements(behavior.getOwnedParameters());
337 	
338 		for (IASTNode declaratorChild : declarator.getChildren()) {
339 			if (declaratorChild instanceof IASTParameterDeclaration) {
340 				IASTParameterDeclaration parameter = (IASTParameterDeclaration) declaratorChild;
341 				IASTName parameterName = parameter.getDeclarator().getName();
342 				IASTDeclSpecifier parameterType = parameter.getDeclSpecifier();
343 				ParameterModifiers modifiers = new ParameterModifiers();
344 				String parameterTypeName = ""; //$NON-NLS-1$
345 				try {
346 					IToken token = parameter.getDeclarator().getSyntax();
347 					while (token != null) {
348 						String tokenStr = token.toString();
349 						if (tokenStr.equals("*")) { //$NON-NLS-1$
350 							modifiers.isPointer = true;
351 						} else if (tokenStr.equals("&")) { //$NON-NLS-1$
352 							modifiers.isRef = true;
353 						} else if (tokenStr.equals("[")) { //$NON-NLS-1$
354 							while (token != null) {
355 								modifiers.array += token.toString();
356 								token = token.getNext();
357 							}
358 							if (token == null) {
359 								break;
360 							}
361 						}
362 						token = token.getNext();
363 					}
364 
365 					token = parameterType.getSyntax();
366 					while (token != null) {
367 						String tokenStr = token.toString();
368 						if (tokenStr.equals("*")) { //$NON-NLS-1$
369 							// TODO: check, if this can be called (depending on
370 							// * position with different semantics?)
371 							modifiers.isPointer = true;
372 						} else if (tokenStr.equals("&")) { //$NON-NLS-1$
373 							modifiers.isRef = true;
374 						} else if (tokenStr.equals(REGISTER)) {
375 							modifiers.isRegister = true;
376 						} else if (tokenStr.equals(CONST) || tokenStr.equals(VOLATILE)) {
377 							// do nothing (use isConst() or isVolatile() operation of parameterType)
378 							// is not part of parameter type
379 						} else {
380 							if (parameterTypeName.length() > 0) {
381 								parameterTypeName += " "; //$NON-NLS-1$
382 							}
383 							parameterTypeName += tokenStr;
384 						}
385 						token = token.getNext();
386 					}
387 				} catch (ExpansionOverlapsBoundaryException e) {
388 				}
389 
390 				NamedElement namedElemParamType = Utils.getQualifiedElement(Utils.getTop(m_classifier),
391 						parameterTypeName);
392 				if (namedElemParamType == null) {
393 					String qParName = ansiCLib + Utils.nsSep + parameterTypeName;
394 					namedElemParamType = ElementUtils.getQualifiedElementFromRS(m_classifier, qParName);
395 					if (namedElemParamType == null) {
396 						// Make sure that Ansi-C library is loaded
397 						StdModelLibs.addResource(CppUriConstants.ANSIC_LIB, m_classifier);
398 						namedElemParamType = ElementUtils.getQualifiedElementFromRS(m_classifier, qParName);
399 					}
400 					// still null? recover type from existing definition (provided that name has not changed)
401 					if (namedElemParamType == null) {
402 						for (Parameter existingParameter : existingParameters) {
403 							if (existingParameter.getName().equals(parameterName.toString())) {
404 								namedElemParamType = existingParameter.getType();
405 								break;
406 							}
407 						}
408 					}
409 				}
410 				if (parameterType.isRestrict()) {
411 				}
412 				Parameter umlParameter = null;
413 				Type paramType = namedElemParamType instanceof Type ? (Type) namedElemParamType : null;
414 				if (operation != null) {
415 					umlParameter = operation.createOwnedParameter(parameterName.toString(), paramType);
416 					applyParameterModifiers(parameterType, umlParameter, modifiers);
417 				}
418 				umlParameter = behavior.createOwnedParameter(parameterName.toString(), paramType);
419 				applyParameterModifiers(parameterType, umlParameter, modifiers);
420 			}
421 		}
422 
423 		if (behavior instanceof OpaqueBehavior) {
424 			OpaqueBehavior ob = (OpaqueBehavior) behavior;
425 			if (ob.getBodies().size() == 0) {
426 				ob.getLanguages().add(c_cpp_langID);
427 				ob.getBodies().add(""); //$NON-NLS-1$
428 			}
429 			for (int i = 0; i < ob.getLanguages().size(); i++) {
430 				// update first body of one of the languages supported by CDT. This implies that
431 				// it is actually not possible to have separate C and C++ bodes in the same opaque
432 				// behavior (which is rarely a good idea).
433 				String language = ob.getLanguages().get(i);
434 				if (TextEditorConstants.CPP.matcher(language).matches() || c_cpp_langID.equals(language)) {
435 					if (i < ob.getBodies().size()) {
436 						// should always be true, unless sync between
437 						// languages/bodies is lost
438 						ob.getBodies().set(i, body);
439 					}
440 				}
441 			}
442 		}
443 		if (operation != null) {
444 			return operation;
445 		}
446 		else {
447 			return behavior;
448 		}
449 	}

451 	/**
452 	 * Apply the modifiers for a parameter, notably the stereotypes of the C++ profile
453 	 * 
454 	 * @param parameterType the CDT AST parameter specification
455 	 * @param umlParameter the UML parameter (to which a stereotype should be applied)
456 	 * @param modifiers the modifiers that should be applied (stored in an instance of class ParameterModifiers)
457 	 */
458 	public void applyParameterModifiers(IASTDeclSpecifier parameterType, Parameter umlParameter, ParameterModifiers modifiers) {
459 		if (parameterType.isConst()) {
460 			StereotypeUtil.apply(umlParameter, Const.class);
461 		}
462 		if (parameterType.isVolatile()) {
463 			StereotypeUtil.apply(umlParameter, Volatile.class);
464 		}
465 		if (modifiers.isRegister) {
466 			StorageClass sc = StereotypeUtil.applyApp(umlParameter, StorageClass.class);
467 			if (sc != null) {
468 				sc.setStorageClass(EStorageClass.REGISTER);
469 			}
470 		}
471 		if (modifiers.isPointer) {
472 			StereotypeUtil.apply(umlParameter, Ptr.class);
473 		} else if (modifiers.isRef) {
474 			StereotypeUtil.apply(umlParameter, Ref.class);
475 		}
476 		if (modifiers.array.length() > 0) {
477 			Array arraySt = StereotypeUtil.applyApp(umlParameter, Array.class);
478 			if (arraySt != null && !modifiers.array.equals("[]") && (!modifiers.array.equals("[ ]"))) { //$NON-NLS-1$//$NON-NLS-2$
479 				arraySt.setDefinition(modifiers.array);
480 			}
481 		}
482 	}

484 	/**
485 	 * Obtain an operation from the model by using the name of a CDT method.
486 	 * If an operation of the given name does not exist, it might indicate that
487 	 * the method has been renamed.
488 	 * 
489 	 * @param name the operation name within CDT
490 	 * @param parent the parent of the CDT method within CDT editor model
491 	 * @param position the position within the other methods. This information is used to locate methods
492 	 *   within the model that might have been renamed in the CDT editor.
493 	 * @return
494 	 */
495 	public Operation getModelOperationFromName(String name, IParent parent, int position) {
496 		Operation operation = m_classifier.getOperation(name, null, null);
497 		
498 		if (operation == null) {
499 			// operation is not found via name in the model. try to locate the operation in the model at the same
500 			// "position" as the method in the file and
501 			// verify that this method does not have the same name as any method
502 			// in the CDT file.
503 			if (position < m_classifier.getOperations().size()) {
504 				operation = m_classifier.getOperations().get(position);
505 				String modelName = operation.getName();
506 				try {
507 					for (ICElement child : parent.getChildren()) {
508 						if (child instanceof IMethodDeclaration) {
509 							String cdtName = ((IMethodDeclaration) child).getElementName();
510 							if (cdtName.equals(modelName)) {
511 								// an existing operation in the CDT file already
512 								// has this name
513 								operation = null;
514 								break;
515 							}
516 						}
517 					}
518 				} catch (CModelException e) {
519 				}
520 			}
521 		}
522 		return operation;
523 	}

525 	public static String getBody(ITranslationUnit itu, IASTFunctionDefinition definition) {
526 		IASTStatement body = definition.getBody();
527 
528 		if (body instanceof IASTCompoundStatement) {
529 			IASTCompoundStatement bodyComp = (IASTCompoundStatement) body;
530 
531 			IASTFileLocation bodyLoc = bodyComp.getFileLocation();
532 			int start = bodyLoc.getNodeOffset();
533 			int end = start + bodyLoc.getNodeLength();
534 			char contents[] = itu.getContents();
535 			// body contains enclosing { } which we need to remove (+2, -2). We
536 			// cannot use the
537 			// first and last statement, since leading and trailing comments are
538 			// not part of the AST tree.
539 			return Utils.decreaseIndent(contents, start + 2, end - 2);
540 		}
541 		return ""; //$NON-NLS-1$
542 	}
543 
544 	/**
545 	 * update a comment of a named element. Besides the comment of the element itself, comments on contained
546 	 * parameters are handled.
547 	 * 
548 	 * @param itu a translation unit
549 	 * @param definition
550 	 * @param ne a named element that is either an operation or a behavior (in order to update parameters)
551 	 */
552 	public void updateComment(ITranslationUnit itu, IASTFunctionDefinition definition, NamedElement ne) {
553 		IASTFileLocation bodyLoc = definition.getFileLocation();
554 		int start = bodyLoc.getNodeOffset() - 1;
555 		int end = start;
556 		char contents[] = itu.getContents();
557 		String comment = ""; //$NON-NLS-1$
558 		// backward scan for beginning /*
559 		while (start > 0) {
560 			if (contents[start] == '/' && contents[start + 1] == '*') {
561 				start += "/**".length(); // TODO: common string //$NON-NLS-1$
562 											// constants with generator
563 				for (int i = start; i < end; i++) {
564 					comment += contents[i];
565 				}
566 				comment = comment.replace("\n * ", "\n"). //$NON-NLS-1$//$NON-NLS-2$
567 						replace("*/", "").trim(); //$NON-NLS-1$//$NON-NLS-2$
568 				break;
569 			}
570 			start--;
571 		}
572 		if (comment.length() > 0) {
573 			// filter @param
574 			int atParam = comment.indexOf(sAtParam);
575 			String commentMethodOnly = (atParam != -1) ? comment.substring(0, atParam).trim() : comment;
576 
577 			EList<Comment> commentsUML = ne.getOwnedComments();
578 			Comment commentUML;
579 			if (commentsUML.size() == 0) {
580 				commentUML = ne.createOwnedComment();
581 				commentUML.getAnnotatedElements().add(commentUML);
582 			} else {
583 				commentUML = commentsUML.get(0);
584 			}
585 			while (atParam != -1) {
586 				int currentAtParam = atParam;
587 				atParam = comment.indexOf(sAtParam, atParam + 1);
588 				String commentParam = (atParam != -1) ? comment.substring(currentAtParam, atParam)
589 						: comment.substring(currentAtParam);
590 				Comment commentParamUML;
591 				int atParamName = sAtParam.length();
592 
593 				while ((atParamName < commentParam.length())
594 						&& Character.isWhitespace(commentParam.charAt(atParamName))) {
595 					atParamName++;
596 				}
597 				int atParamNameEnd = atParamName;
598 				while ((atParamNameEnd < commentParam.length())
599 						&& !Character.isWhitespace(commentParam.charAt(atParamNameEnd))) {
600 					atParamNameEnd++;
601 				}
602 				if (atParamNameEnd < commentParam.length() - 1) {
603 					String parameterName = commentParam.substring(atParamName, atParamNameEnd);
604 					String commentParamText = commentParam.substring(atParamNameEnd).trim();
605 					Parameter parameter = null;
606 					if (ne instanceof BehavioralFeature) {
607 						parameter = ((BehavioralFeature) ne).getOwnedParameter(parameterName, null, false, false);
608 					}
609 					else if (ne instanceof Behavior) {
610 						parameter = ((Behavior) ne).getOwnedParameter(parameterName, null, false, false);		
611 					}
612 					if (parameter != null) {
613 						EList<Comment> commentsParamUML = parameter.getOwnedComments();
614 						if (commentsParamUML.size() == 0) {
615 							commentParamUML = parameter.createOwnedComment();
616 							commentParamUML.getAnnotatedElements().add(commentParamUML);
617 						} else {
618 							commentParamUML = commentsParamUML.get(0);
619 						}
620 						commentParamUML.setBody(commentParamText);
621 					} else {
622 						// parameter is not found in model, e.g. either renamed
623 						// or not yet existing
624 						// store comment in operation comment
625 						commentMethodOnly += "\n " + sAtParam + parameterName + //$NON-NLS-1$
626 								" not found(!) " + commentParamText; //$NON-NLS-1$
627 					}
628 				}
629 			}
630 			commentUML.setBody(commentMethodOnly);
631 		}
632 	}
