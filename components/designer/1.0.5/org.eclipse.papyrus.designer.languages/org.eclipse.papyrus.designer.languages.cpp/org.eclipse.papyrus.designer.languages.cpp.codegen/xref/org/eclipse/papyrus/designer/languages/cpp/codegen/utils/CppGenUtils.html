63  	/**
64  	 * Return a list of template parameters without type
65  	 *
66  	 * @param classifier
67  	 * @return
68  	 */
69  	public static String getTemplateParametersWoType(Classifier classifier) {
70  		String tparamWoType = ""; //$NON-NLS-1$
71  
72  		Iterator<TemplateParameter> it = GenUtils.getTemplateParameters(classifier).iterator();
73  
74  		while (it.hasNext()) {
75  			TemplateParameter currentTParam = it.next();
76  			tparamWoType = tparamWoType + GenUtils.getTemplateName(currentTParam);
77  
78  			if (it.hasNext()) {
79  				tparamWoType = tparamWoType + ", "; //$NON-NLS-1$
80  			}
81  		}
82  		return tparamWoType;
83  	}

85  	/**
86  	 * The standard UML and MARTE libraries do not apply the stereotype "Typedef". Yet, we want to treat these
87  	 * types in an identical way, i.e. we use a typedef to the associated primitive C++ type
88  	 *
89  	 * @param type
90  	 * @return
91  	 */
92  	public static String getStdtypes(PrimitiveType type) {
93  		Object owner = type.getOwner();
94  		String owningPkgName = ""; //$NON-NLS-1$
95  		if (owner instanceof Package) {
96  			owningPkgName = ((Package) owner).getName();
97  		}
98  		if (owningPkgName.equals("PrimitiveTypes") || // used in UML >= 2.4 //$NON-NLS-1$
99  				owningPkgName.equals("UMLPrimitiveTypes") || // used in UML < 2.4 //$NON-NLS-1$
100 				owningPkgName.equals("MARTE_PrimitivesTypes")) { //$NON-NLS-1$
101 			String td = null;
102 			String name = type.getName();
103 
104 			if (name.equals("Boolean")) { //$NON-NLS-1$
105 				td = "bool"; //$NON-NLS-1$
106 			}
107 			else if (name.equals("Integer")) { //$NON-NLS-1$
108 				td = "int"; //$NON-NLS-1$
109 			}
110 			else if (name.equals("Unlimited Natural")) { //$NON-NLS-1$
111 				td = "unsigned long"; //$NON-NLS-1$
112 			}
113 			else if (name.equals("Real")) { //$NON-NLS-1$
114 				td = "float"; //$NON-NLS-1$
115 			}
116 			else if (name.equals("String")) { //$NON-NLS-1$
117 				td = "const char *"; //$NON-NLS-1$
118 			}
119 			else if (name.equals("Unlimited Natural")) { //$NON-NLS-1$
120 				td = "unsigned long"; //$NON-NLS-1$
121 			}
122 			if (td != null) {
123 				return "typedef " + td + " " + name + ";"; //$NON-NLS-1$ //$NON-NLS-2$ //$NON-NLS-3$
124 			}
125 			// else unknown within UMLPrimitiveTypes, treat in standard way
126 		}
127 
128 		return ""; //$NON-NLS-1$
129 	}

131 	/**
132 	 * Specific handling of CORBA basic types. These have the property that the
133 	 * type name itself differs from the declared primitive type (in the C++ mapping)
134 	 * TODO: This should be handled by a modular extension defined in CORBA modules.
135 	 *
136 	 * @param type
137 	 *            A primitive type
138 	 * @return a new type name if for CORBA, null otherwise
139 	 */
140 	public static String transformCORBABaseTypes(PrimitiveType type) {
141 		Object owner = type.getOwner();
142 		String owningPkgName = ""; //$NON-NLS-1$
143 		if (owner instanceof Package) {
144 			owningPkgName = ((Package) owner).getName();
145 		}
146 		if (owningPkgName.equals("CORBA")) { //$NON-NLS-1$
147 			String cppName = "CORBA::"; //$NON-NLS-1$
148 			boolean nextUpper = true; // first char is upper
149 			for (char ch : type.getName().toCharArray()) {
150 				if (ch == ' ') { // after " " is upper
151 					nextUpper = true;
152 				} else {
153 					if (nextUpper) {
154 						cppName += Character.toUpperCase(ch);
155 					} else {
156 						cppName += ch;
157 					}
158 					nextUpper = false;
159 				}
160 			}
161 			return cppName;
162 		}
163 		return null;
164 	}

166 	/**
167 	 * Return a kind of qualifiedName, except if
168 	 * - The named element has the stereotype External or NoCodeGen
169 	 * - The named element is part of the ANSI C library
170 	 * - The named element is a primitive type that has no further definition via a stereotype (TODO: why is this required/useful?)
171 	 *
172 	 * @param ne
173 	 * @return
174 	 */
175 	public static String cppQualifiedName(NamedElement ne) {
176 		if (ne == null) {
177 			return "undefined"; //$NON-NLS-1$
178 		}
179 		Object owner = ne.getOwner();
180 		String owningPkgName = ""; //$NON-NLS-1$
181 
182 		if (owner instanceof Package) {
183 			owningPkgName = ((Package) owner).getName();
184 		}
185 
186 		if (ne instanceof PrimitiveType) {
187 			// is a standard type
188 			String cppType = transformCORBABaseTypes((PrimitiveType) ne);
189 			if (cppType != null) {
190 				return cppType;
191 			}
192 		}
193 
194 		if (GenUtils.hasStereotypeTree(ne, NoCodeGen.class)) {
195 			return ne.getName();
196 		}
197 		else if (GenUtils.hasStereotypeTree(ne, ExternLibrary.class)) {
198 			ExternLibrary extLibrary = GenUtils.getApplicationTree(ne, ExternLibrary.class);
199 			// handle potential prefix defined for all types within the external library
200 			return GenUtils.maskNull(extLibrary.getPrefix()) + ne.getName();
201 		}
202 		else if (GenUtils.hasStereotype(ne, External.class)) {
203 			// check External after ExternLibrary stereotype, in order to take an
204 			// eventual prefix defined by the latter into account
205 			return ne.getName();
206 		} 
207 		else if (owningPkgName.equals(ANSI_C_LIB)) {
208 			// always use the short name for types within the ANSI C library
209 			return ne.getName();
210 		}
211 		else if (owner instanceof ClassifierTemplateParameter) {
212 			// return short name for template in Type
213 			return ne.getName();
214 		}
215 
216 		String qName = ne.getName();
217 		if (currentNS == ne.getNamespace()) {
218 			// return simple name, if in current namespace
219 			return qName;
220 		}
221 		if (ne instanceof PrimitiveType) {
222 			if (!GenUtils.hasStereotype(ne, Typedef.class) && (getStdtypes((PrimitiveType) ne).length() == 0)) {
223 				// is a primitive type without further definition and not a standard primitive type
224 				// => assume that it is a external type without namespace
225 				return qName;
226 			}
227 		}
228 
229 		for (Namespace ns : ne.allNamespaces()) {
230 			if (ns.getOwner() != null || USE_ALL_NS) {
231 				qName = ns.getName() + NamedElement.SEPARATOR + qName;
232 			}
233 		}
234 		if (qName == null) {
235 			GenUtils.checkProxy(ne);
236 			return "undefined"; //$NON-NLS-1$
237 		}
238 		if (qName.contains(NamedElement.SEPARATOR)) {
239 			// is a qualified name => make path absolute
240 			return NamedElement.SEPARATOR + qName;
241 		} else {
242 			return qName;
243 		}
244 	}

246 	/**
247 	 * Returns the string that is used within a C++ template declaration, e.g. the "Class XY" in template<class XY>.
248 	 *
249 	 * @return the template type formated as string
250 	 */
251 	public static String getTemplateTypeName(TemplateParameter templateParam) {
252 		String prefix = ""; //$NON-NLS-1$
253 		String name = ""; //$NON-NLS-1$
254 
255 		// Retrieve name of the ParameteredElement (when possible = it is a NamedElement
256 		ParameterableElement pElt = templateParam.getParameteredElement();
257 		if ((pElt != null) && (pElt instanceof NamedElement)) {
258 			name = ((NamedElement) pElt).getName();
259 			if (templateParam instanceof ClassifierTemplateParameter) {
260 				// prefix name with "class" to indicate that the following type is a classifier
261 				prefix = "class "; //$NON-NLS-1$
262 			} else if (templateParam instanceof OperationTemplateParameter) {
263 				// no prefix
264 			} else if (templateParam instanceof ConnectableElementTemplateParameter) {
265 				// no prefix
266 			} else {// uml2TemplateParameter instanceof TemplateParameter
267 				// TODO: literal integer is a value specification, but we use the fact that is also a named element
268 				// e.g. would produce an <int N> parameter and ignore the value behind (which is ok for the
269 				// declaration of the template, but not for its use.
270 				if (pElt instanceof LiteralInteger) {
271 					prefix = "int "; //$NON-NLS-1$
272 				} else if (pElt instanceof LiteralString) {
273 					prefix = "String "; //$NON-NLS-1$
274 				} else if (pElt instanceof LiteralBoolean) {
275 					prefix = "Boolean "; //$NON-NLS-1$
276 				} else if (pElt instanceof LiteralUnlimitedNatural) {
277 					prefix = "UnlimitedNatural "; //$NON-NLS-1$
278 				} else {
279 					prefix = pElt.eClass().getName() + " "; //$NON-NLS-1$
280 				}
281 			}
282 		} else {
283 			name = "undefined"; //$NON-NLS-1$
284 		}
285 
286 		return (prefix + name);
287 	}

291 	 * Return a C++ namespace definition for a named element
292 	 *
293 	 * @param ne
294 	 *            a named element
295 	 * @return a C++ namespace definition for a named element
296 	 */
297 	public static String getNamespace(NamedElement ne) {
298 		String namespace = ""; //$NON-NLS-1$
299 		for (Namespace ns : ne.allNamespaces()) {
300 			if (ns.getOwner() != null) {
301 				String nsName = ns.getName();
302 				if (!namespace.equals("")) { //$NON-NLS-1$
303 					nsName += NamedElement.SEPARATOR;
304 				}
305 				namespace = nsName + namespace;
306 			}
307 		}
308 		if (!namespace.equals("")) { //$NON-NLS-1$
309 			namespace = "\n" + "using namespace " + namespace + ";\n"; //$NON-NLS-1$ //$NON-NLS-2$ //$NON-NLS-3$
310 		}
311 		return namespace;
312 	}

314 	/**
315 	 * Return all namespaces, with specific case of a package (which needs to
316 	 * include itself as a namespace unlike other elements (e.g. classes)
317 	 */
318 	public static EList<Namespace> getAllNamespaces(NamedElement ne) {
319 		if (ne instanceof Package) {
320 			EList<Namespace> allNamespaces = new BasicEList<Namespace>();
321 			allNamespaces.add((Package) ne);
322 			allNamespaces.addAll(ne.allNamespaces());
323 			return allNamespaces;
324 		}
325 		else {
326 			return ne.allNamespaces();
327 		}
328 	}

331 	 * Return a C++ open-namespace definition for a named element
332 	 *
333 	 * @param ne
334 	 *            a named element
335 	 * @return a C++ open-namespace definition for a named element
336 	 */
337 	public static String openNS(NamedElement ne) {
338 		String openNS = ""; //$NON-NLS-1$
339 		currentNS = ne.getNamespace();
340 		for (Namespace ns : getAllNamespaces(ne)) {
341 			if (ns.getOwner() != null || USE_ALL_NS) {
342 				openNS = "namespace " + ns.getName() + " {\n" + openNS; //$NON-NLS-1$ //$NON-NLS-2$
343 			}
344 		}
345 		return openNS;
346 	}

420 	private static VisibilityKind currVisibility = null;
421 	private static final Pattern EmptySectionRegex = Pattern.compile("^\\s*$"); //$NON-NLS-1$
422 
423 	/**
424 	 * Update the current visibility to the specified value without writing this value to
425 	 * the output. This is to be used when setting the default visibility of a class/struct.
426 	 */
427 	public static void resetVisibility(VisibilityKind v) {
428 		currVisibility = v;
429 	}
430 
431 	/**
432 	 * Get the current visibility so we can restore it when generating nested classes
433 	 */
434 	public static VisibilityKind getCurrentVisibility() {
435 		return currVisibility;
436 	}
437 
438 	/**
439 	 * Create a section of code with the appropriate visibility. Merges the content with
440 	 * the previously declared visibility (if appropriate). Ignore empty content.
441 	 */
442 	public static String getSection(VisibilityKind visibility, String content) {
443 		// Bug 425208: Don't update the visibility until we know for sure that it
444 		// will be written to the output.
445 		VisibilityKind newVisibility = null;
446 		if (!visibility.equals(currVisibility)) {
447 			newVisibility = visibility;
448 		}
449 
450 		// Filter out empty sections.
451 		Matcher m = EmptySectionRegex.matcher(content);
452 		if (content.isEmpty() || m.matches()) {
453 			return ""; //$NON-NLS-1$
454 		}
455 
456 		// Don't write duplicate visibility modifiers.
457 		if (newVisibility == null) {
458 			return content;
459 		}
460 
461 		currVisibility = newVisibility;
462 		return currVisibility.toString() + ":\n" + content; //$NON-NLS-1$
463 	}
