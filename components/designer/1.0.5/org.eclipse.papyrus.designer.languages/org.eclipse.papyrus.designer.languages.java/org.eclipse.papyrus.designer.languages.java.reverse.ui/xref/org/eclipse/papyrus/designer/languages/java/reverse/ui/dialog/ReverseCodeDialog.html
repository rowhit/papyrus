101 	public ReverseCodeDialog(Shell parentShell, String modelUid, String projectName, List<String> searchPathsInitialValues) {
102 		super(parentShell, dialogTitle, textMsg, getInitialValue(modelUid, projectName), null);
103 		// TODO Auto-generated constructor stub
104 		IDialogSettings settings = Activator.getDefault().getDialogSettings();
105 
106 		MODEL_UID = modelUid;
107 		this.projectName = projectName;
108 
109 		// Look for generationPackageName if none is provided.
110 		// if(initialValue == null)
111 		// {
112 		// String generationPackageName = settings.get("generationPackageName");
113 		// getText().setText(generationPackageName);
114 		//
115 		// }
116 		
117 		// Find default values for search path and creation path
118 		ReversePreference preference = new ReversePreference();
119 		
120 		if (searchPathsInitialValues == null) {
121 			String[] savedSearchPath = settings.getArray( getRootSettingKey() + SEARCHPATHS_UID);
122 			if (savedSearchPath != null) {
123 				searchPathsInitialValues = Arrays.asList(savedSearchPath);
124 			} else
125 			{
126 				searchPathsInitialValues = getDefaultSearchPath(preference);
127 			}
128 		}
129 
130 		// Look for saved creationPaths if none is provided.
131 		if (creationPaths == null) {
132 			String[] savedSearchPath = settings.getArray( getRootSettingKey() + CREATIONPATHS_UID);
133 			if(savedSearchPath != null)
134 				creationPaths = savedSearchPath;
135 			else
136 				creationPaths = getDefaultCreationPath(preference);
137 			}
138 
139 		// Look for saved creationPaths if none is provided.
140 		if(creationPaths == null ) {
141 			String[] savedSearchPath = settings.getArray( getRootSettingKey() + CREATIONPATHS_UID);
142 			if (savedSearchPath != null) {
143 				creationPaths = savedSearchPath;
144 			} else {
145 				creationPaths = CreationPackageCatalog.getDefaultPackageCreationPatterns(" ; ");
146 			}
147 		}
148 
149 		listDialog = new InputListDialog(listMsg, searchPathsInitialValues);
150 		listDialog.setTooltips(listTooltips);
151 		creationPathsDialog = new InputListDialog(creationPathMsg, Arrays.asList(creationPaths));
152 		creationPathsDialog.setTooltips(creationPathTooltips);
153 		
154 		// Selected parser
155 		try {
156 			selectedParserIndex = settings.getInt(getRootSettingKey() + SELECTEDPARSER_UID);
157 		} catch (NumberFormatException e) {
158 			// Use default parser
159 			selectedParserIndex = 0;
160 		}
161 	}

274 	/**
275 	 * Save the searchpath after the button is pressed.
276 	 */
277 	@Override
278 	protected void okPressed() {
279 		// Save the list before the control is disposed
280 		searchPath = listDialog.getList();
281 		creationPaths = creationPathsDialog.getList();
282 		String defaultCreationPath = getValue();
283 		if (defaultCreationPath == null || defaultCreationPath.length() == 0) {
284 			defaultCreationPath = CreationPackageCatalog.getDefaultCreationPath();
285 		}
286 
287 		// Check inputs
288 		try {
289 			// Check paths
290 			splittedCreationPaths = computeCreationPaths(creationPaths);
291 			CreationPackageCatalog.validateCreationPath(splittedCreationPaths);
292 			// add default searchPaths
293 			searchPath = addDefaultSearchPaths(Arrays.asList(searchPath), splittedCreationPaths).toArray(new String[0]);
294 
295 		} catch (Exception e) {
296 			// TODO Auto-generated catch block
297 			showError(e.getMessage());
298 			return;
299 		}
300 		
301 		// Get selected parser
302 		selectedParserIndex = parserSelectorCombo.getSelectionIndex();
303  
304 		
305 		// save values
306 		IDialogSettings settings = Activator.getDefault().getDialogSettings();
307 		settings.put(getRootSettingKey() + SEARCHPATHS_UID, searchPath);
308 		settings.put(getRootSettingKey() + CREATIONPATHS_UID, creationPaths);
309 		settings.put(getRootSettingKey() + PROJECTNAME_UID, getValue());
310 		settings.put(getRootSettingKey() + SELECTEDPARSER_UID, selectedParserIndex);
311 
312 		super.okPressed();
313 	}

316 	/**
317 	 * Add the defaultSearchPath extracted from the splittedCreationPath to the searchPath.
318 	 *
319 	 * @param splittedCreationPaths2
320 	 */
321 	private List<String> addDefaultSearchPaths(List<String> searchPath, List<String> splittedCreationPaths) {
322 
323 
324 		List<String> result = new ArrayList<String>(searchPath.size() + splittedCreationPaths.size());
325 		result.addAll(searchPath);
326 
327 		List<String> additionalPaths = CreationPackageCatalog.extractCreationPaths(splittedCreationPaths.toArray(new String[0]));
328 		for (String toAdd : additionalPaths) {
329 			if (!result.contains(toAdd)) {
330 				result.add(toAdd);
331 			}
332 		}
333 
334 		return result;
335 	}

348 	/**
349 	 * Compute and check the creationsPaths
350 	 *
351 	 * @param creationPaths
352 	 *            An array with one creationPattern by element
353 	 * @return An array with one path by elements
354 	 * @throws Exception
355 	 */
356 	private List<String> computeCreationPaths(String[] creationPaths) throws Exception {
357 
358 		List<String> result = new ArrayList<String>();
359 
360 		// Iterate line by line
361 		// For each line, separate the 3 patterns
362 		// If a pattern is empty, put a null
363 		for (String line : creationPaths) {
364 			String[] eles = line.split(";");
365 			if (eles.length != 3) {
366 				throw new Exception("A line must contains 3 pattern separated by ';' :" + line);
367 			}
368 			for (String ele : eles)
369 			{
370 				ele = ele.trim();
371 				if (ele.length() == 0) {
372 					ele = null;
373 				}
374 				// add ele
375 				result.add(ele);
376 			}
377 		}
378 		return result;
379 	}
