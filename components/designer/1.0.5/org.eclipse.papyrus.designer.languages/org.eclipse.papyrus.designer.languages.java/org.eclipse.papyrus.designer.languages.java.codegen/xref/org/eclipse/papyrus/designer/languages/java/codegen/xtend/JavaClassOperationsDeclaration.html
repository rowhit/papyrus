13  import java.util.Collection;
14  import org.eclipse.papyrus.designer.languages.common.base.GenUtils;
15  import org.eclipse.papyrus.designer.languages.java.codegen.xtend.JavaOperations;
16  import org.eclipse.papyrus.designer.languages.java.profile.PapyrusJava.NoCodeGen;
17  import org.eclipse.uml2.uml.Classifier;

20  
21  @SuppressWarnings("all")
22  public class JavaClassOperationsDeclaration {
23    public static CharSequence javaClassOperationsDeclaration(final Classifier clazz) {
24      StringConcatenation _builder = new StringConcatenation();
25      {
26        Collection<Operation> _ownedOperations = JavaOperations.getOwnedOperations(clazz);
27        for(final Operation op : _ownedOperations) {
28          {
29            boolean _hasStereotype = GenUtils.hasStereotype(op, NoCodeGen.class);
30            boolean _not = (!_hasStereotype);
31            if (_not) {
32              CharSequence _javaOperationDeclaration = JavaOperations.javaOperationDeclaration(op);
33              _builder.append(_javaOperationDeclaration, "");
34            }
35          }
36          _builder.newLineIfNotEmpty();
37        }
38      }
39      return _builder;
40    }
41  }
