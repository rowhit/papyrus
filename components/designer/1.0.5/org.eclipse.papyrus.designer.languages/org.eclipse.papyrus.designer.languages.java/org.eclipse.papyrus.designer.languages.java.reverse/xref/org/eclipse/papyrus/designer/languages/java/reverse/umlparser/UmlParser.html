21  public class UmlParser {

23  	/**
24  	 * The visitor used to collect model.
25  	 */
26  	protected CompilationUnitAnalyser visitor;
27  
28  	public UmlParser() {
29  	}
30  
31  	/**
32  	 * Select the appropriate parsing method according to the file type.
33  	 *
34  	 * @param file
35  	 * @throws ParseException
36  	 * @throws IOException
37  	 */
38  	public void parseResource(File file) throws ParseException, IOException {
39  		if (file.isDirectory()) {
40  			System.out.println("  '" + file.getAbsolutePath() + " is a directory");
41  			parsePackage(file);
42  		} else if (file.isFile()) {
43  			System.out.println("  '" + file.getAbsolutePath() + " is a file");
44  			// Parse only file ending in .java
45  			if (file.getName().endsWith(".java")) {
46  				parseFile(file);
47  			} else {
48  				System.out.println("skip non java file '" + file.getAbsolutePath() + "'");
49  			}
50  		} else {
51  			System.out.println("  '" + file.getAbsolutePath() + " can't be identified - " + file);
52  		}
53  	}
54  
55  	/**
56  	 * Explore the package to call parse on classes
57  	 *
58  	 * @param file
59  	 * @throws ParseException
60  	 * @throws IOException
61  	 */
62  	public void parsePackage(File file) throws ParseException, IOException {
63  		for (File f : file.listFiles()) {
64  			parseResource(f);
65  		}
66  	}
67  
68  	/**
69  	 * Parse the specified file
70  	 *
71  	 * @param filename
72  	 * @throws IOException
73  	 * @throws ParseException
74  	 */
75  	protected void parseFile(File file) throws ParseException, IOException {
76  		CompilationUnit cu = JavaParser.parse(file);
77  		// prints the resulting compilation unit to default system output
78  		// System.out.println(cu.toString());
79  
80  		// visitor.visit(cu, null);
81  
82  		visitor.processCompilationUnit(cu);
83  	}
84  
85  	/**
86  	 * Select the appropriate parse method according to the type of the
87  	 * resource denoted by the name.
88  	 *
89  	 * @param resourceName
90  	 * @throws IOException
91  	 * @throws ParseException
92  	 */
93  	public void parseResource(String resourceName) throws ParseException, IOException {
94  		File file = new File(resourceName);
95  		parseResource(file);
96  	}
97  
98  	public void execute(String resourcePath, String outputModelName) throws ParseException, IOException {
99  		System.out.println("resource=" + resourcePath);
100 		// Load model
101 		ModelManager modelManager = new ModelManager();
102 		modelManager.createModel(outputModelName);
103 		// modelManager.getModel().load(null);
104 		// Create visitor
105 		visitor = new CompilationUnitAnalyser(modelManager.getModel());
106 		// parse all file using the visitor
107 		parseResource(resourcePath);
108 
109 		// Save the model
110 		modelManager.save();
111 
112 	}
