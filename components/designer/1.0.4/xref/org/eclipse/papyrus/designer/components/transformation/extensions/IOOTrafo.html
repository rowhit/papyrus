14  package org.eclipse.papyrus.designer.components.transformation.extensions;
15  
16  import org.eclipse.papyrus.designer.transformation.base.utils.TransformationException;
17  import org.eclipse.papyrus.designer.transformation.core.transformations.LazyCopier;
18  import org.eclipse.uml2.uml.Class;
19  
20  /**
21   * An interface that must be implemented by classes that realize the transformation
22   * from component-based to object-oriented models. It includes the replacement of
23   * ports and connectors. Ports are
24   * replaced with attributes and access operations, connectors within a composite
25   * by an operation that creates the initial setup.
26   * 
27   * 1. add an operation that allows to retrieve the reference to an interface provided
28   * by a port. This operation has a mapping specific name, e.g. get_<port_name>
29   * 2. add an operation that allows to connect a specific port.
30   * the connect_q operation (*including a
31   * storage attribute*) for a port with a required interface
32   * 3. add an implementation for the getcnx_q operation for a port
33   * with a required interface (the operation itself has been added before)
34   *
35   * Problems: need to align bootloader creation with this mapping, since
36   * the bootloader may be responsible for instantiation
37   * 
38   * Caveat: Assure that the folder derivedInterfaces already exists in a model.
39   * Otherwise the call to getProvided/getRequired interface might trigger its
40   * creation resulting in the corruption of list iterators (ConcurrentAccess
41   * exception)
42   *
43   * New considerations: could the OO trafo be seen as a specific form of the
44   * LwContainerTrafo?
45   * [if yes, would that be useful?] Problem: it's not only a merge and operation
46   * interception, but additional operation(s) for ports. But, it might well
47   * be based on similar mechanisms as in merging in a template + Java code instead
48   * of Java code only. Also possible: still have OOTrafo, but there is a collaboration
49   * between the OO trafo and a LW container rule.
50   */
51  public interface IOOTrafo {
52  
53  	/**
54  	 * Initialize the transformation. Pass the copier and the bootloader to the
55  	 * transformation in case it is needed by an implementation of this interface.
56  	 * @param copier an instance of the lazy copier that copies from source (intermediate) model to the OO model.
57  	 * @param bootloader
58  	 */
59  	public void init(LazyCopier copier, Class bootloader);
60  	
61  	/**
62  	 * Transformations adds operations for accessing ports. A common but not necessary
63  	 * code generation pattern is to associate different operations with ports that
64  	 * provide services (access operations) and ports requiring services (connection operations)
65  	 * 
66  	 * @param implementation
67  	 *            A component implementation
68  	 */
69  	public void addPortOperations(Class implementation);
70  
71  
72  	/**
73  	 * Add an operation that connects the parts within a composite class based on the
74  	 * UML connector information.
75  	 *
76  	 * @param compositeImplementation
77  	 *            A (composite) component implementation
78  	 * @throws TransformationException
79  	 */
80  	public void addConnectionOperation(Class compositeImplementation)
81  			throws TransformationException;
82  
83  	/**
84  	 * Transform parts if necessary. This transformation is for target programming languages
85  	 * like C++ that support different ways of composition for attributes, i.e. references/pointers
86  	 * versus values. In case of the latter, the instantiation of a composite directly implies
87  	 * the instantiation of its sub-components. In case of the former, parts need to be instantiated
88  	 * by a different mechanism, e.g. by the constructor (which is still a valid correspondence for
89  	 * composition in UML at a logical level).
90  	 * Another possible use is the support of a more dynamic deployment: it is for instance possible
91  	 * to use a use a part-list (single attribute) instead of fixed named attributes.
92  	 * 
93  	 * @param compositeImplementation
94  	 *            A (composite) component implementation
95  	 */
96  	public void transformParts(Class compositeImplementation);
97  }
