138 		m_component = DepUtils.getImplementation(instance);
139 		m_instance = instance;
140 		m_feature = null;
141 		m_model = PackageUtil.getUserModel(from);
142 		return checkAndGetInstances();

145 	public boolean init(Feature feature, ExecutionEvent from) {
146 		// visitedPackages = new BasicEList<Package> ();
147 		m_feature = feature;
148 		m_model = PackageUtil.getUserModel(from);
149 		if (m_model == null) {
150 			m_model = PackageUtil.getRootPackage(feature);
151 		}
152 		m_instance = null;
153 		// need to overload for connector support
154 		if (feature instanceof Property) {
155 			Type type = ((Property) feature).getType();
156 			if (type instanceof Class) {
157 				m_component = (Class) type;
158 				return checkAndGetInstances();
159 			}
160 		}
161 		return false;
162 	}

164 	/**
165 	 * A filter class that selects instances associated with the selected feature or component
166 	 */
167 	class InstanceFilter implements ElementFilter {
168 
169 		private String featureCandidateName;
170 
171 		@Override
172 		public boolean acceptElement(Element element) {
173 			if (element instanceof InstanceSpecification) {
174 				InstanceSpecification instance = (InstanceSpecification) element;
175 				if (m_feature instanceof Property) {
176 					EList<Slot> slots = DepUtils.getReferencingSlots(instance);
177 					for (Slot slot : slots) {
178 						if (slot.getDefiningFeature() == m_feature) {
179 							return true;
180 						}
181 					}
182 				}
183 				else if (m_feature instanceof Connector) {
184 					// Connector instances cannot be found via a slot. Since a connector is not a structural feature,
185 					// it cannot be referenced via the "definingFeature" property of a slot. Therefore, the deployment
186 					// plan creation (@see DepCreation.createDepPlan) simply omits slots, but still creates instances
187 					// for the interaction components referenced by an FCM connector.
188 					// TODO: the following code relies on specific order of instances and might not always work.
189 					Element owner = m_feature.getOwner();
190 					if (DepUtils.getImplementation(instance) == owner) {
191 						// instance is for owner of feature: examine name
192 						featureCandidateName = instance.getName() + "." + m_feature.getName(); //$NON-NLS-1$
193 					}
194 					else if (featureCandidateName != null) {
195 						if (featureCandidateName.equals(instance.getName())) {
196 							return true;
197 						}
198 					}
199 				} else if (DepUtils.getImplementation(instance) == m_component) {
200 					return true;
201 				}
202 			}
203 			return false;
204 		}
205 	}

208 	/**
209 	 * retrieve the instance list. Returns false, if no deployment plan could be found.
210 	 *
211 	 * @return
212 	 */
213 	public boolean checkAndGetInstances() {
214 		visitedPackages = new BasicEList<Package>();
215 		m_instanceList = new BasicEList<InstanceSpecification>();
216 
217 		ElementFilter filter = new InstanceFilter();
218 		DepUtils.getAllInstances(m_model, m_instanceList, filter);
219 
220 		if (m_instanceList.size() == 0) {
221 			Shell shell = Display.getDefault().getActiveShell();
222 			if (DepPlanUtils.getAllDepPlans(m_model).size() == 0) {
223 				MessageDialog.openInformation(shell, "Instance configuration",
224 						"No deployment plans are defined. Create a deployment plan before configuring instances");
225 				return false;
226 			}
227 			else if (MessageDialog.openConfirm(shell, "Instance configuration",
228 					"The list of available instances is empty. Synchronize deployment plans?")) {
229 				DepPlanSync.syncAllDepPlans(m_model);
230 				visitedPackages = new BasicEList<Package>();
231 				DepUtils.getAllInstances(m_model, m_instanceList, filter);
232 
233 				if (m_instanceList.size() == 0) {
234 					MessageDialog
235 							.openInformation(
236 									shell,
237 									"Instance configuration",
238 									"There are still no instances available\n. Check whether you created already a deployment plan for your system. Check as well, if the parts in your a composite use \"composite\" as aggregation kind. (Results of deployment plan synchronizations will be unrolled)");
239 					return false;
240 				}
241 			}
242 		}
243 		return true;
244 	}

264 	protected void createInstanceSelectionGroup(Composite parent) {
265 		Composite instanceSelection = new Composite(parent, SWT.NONE);
266 		GridLayout grid = new GridLayout(1, true);
267 		GridData groupGridData = new GridData();
268 
269 		groupGridData.grabExcessVerticalSpace = true;
270 		groupGridData.grabExcessHorizontalSpace = true;
271 		groupGridData.horizontalAlignment = GridData.FILL;
272 		groupGridData.verticalAlignment = GridData.FILL;
273 		instanceSelection.setLayout(grid);
274 		instanceSelection.setLayoutData(groupGridData);
275 
276 		/*
277 		 * fRules = DialogUtils.createFilteredList (ruleSelGroup, new
278 		 * RuleLabelProvider(), 200, 200, SWT.BORDER | SWT.V_SCROLL |
279 		 * SWT.H_SCROLL);
280 		 */

282 		//
283 		// --------------- instance selection -------------------
284 		//
285 		Group instanceSelGroup = new Group(instanceSelection, SWT.BORDER);
286 		instanceSelGroup.setText(addSpaces("associated instance specifications"));
287 		// ruleGroup.setLayout(new RowLayout (SWT.VERTICAL));
288 		instanceSelGroup.setLayout(new GridLayout(1, false));
289 		instanceSelGroup.setLayoutData(groupGridData);
290 
291 		fInstanceList = new TreeViewer(instanceSelGroup, SWT.BORDER);
292 
293 		GridData data = new GridData(GridData.FILL_BOTH);
294 		data.heightHint = 100;
295 		data.widthHint = 350;
296 
297 		fInstanceList.getTree().setLayoutData(data);
298 		fInstanceList.setLabelProvider(new InstanceLabelProvider());
299 		fInstanceList.setContentProvider(new ITreeContentProvider() {
300 
301 			@Override
302 			public void inputChanged(Viewer viewer, Object oldInput, Object newInput) {
303 			}
304 
305 			@Override
306 			public void dispose() {
307 			}
308 
309 			@Override
310 			public boolean hasChildren(Object element) {
311 				return getChildren(element).length > 0;
312 			}
313 
314 			@Override
315 			public Object getParent(Object element) {
316 				return null;
317 			}
318 
319 			@Override
320 			public Object[] getElements(Object inputElement) {
321 				return (Object[]) inputElement;
322 			}
323

334 		fInstanceList.addSelectionChangedListener(new ISelectionChangedListener() {

336 			@Override

338 				ISelection selection = fInstanceList.getSelection();
339 				if (selection instanceof StructuredSelection) {
340 					Object[] selected = ((StructuredSelection) selection)

362 	protected void createInstanceConfigurationGroup(Composite parent) {
363 		GridData groupGridData = DialogUtils.createFillGridData();
364 
365 		Group instanceConfigurationGroup = new Group(parent, SWT.BORDER);
366 		instanceConfigurationGroup.setText(addSpaces("Configuration"));
367 		instanceConfigurationGroup.setLayout(new RowLayout(SWT.VERTICAL));
368 		instanceConfigurationGroup.setLayout(new GridLayout(1, false));
369 		instanceConfigurationGroup.setLayoutData(groupGridData);
370 
371 		// fRuleName = new Text (ruleInfoGroup, SWT.NONE);
372 		Label configL = new Label(instanceConfigurationGroup, SWT.NONE);
373 		configL.setText("Available properties:");
374 
375 		fPropertyList = new TableViewer(instanceConfigurationGroup, SWT.BORDER);
376 		fPropertyList.setLabelProvider(new AttributeLabelProvider());
377 		fPropertyList.setContentProvider(new ArrayContentProvider());
378 		fPropertyList.setInput(getConfigAttributes(m_component).toArray());
379 		fPropertyList
380 				.addSelectionChangedListener(new ISelectionChangedListener() {
381 
382 					@Override
383 					public void selectionChanged(SelectionChangedEvent event) {
384 						ISelection selection = fPropertyList.getSelection();
385 						if (selection instanceof StructuredSelection) {
386 							Object[] selected = ((StructuredSelection) selection)
387 									.toArray();
388 							if ((selected.length == 1)
389 									&& (selected[0] instanceof Property)) {
390 								selectProperty((Property) selected[0]);
391 							}
392 						}
393 					}
394 				});
395 
396 		fValueLabel = new Label(instanceConfigurationGroup, SWT.NONE);
397 
398 		GridData span2 = new GridData();
399 		span2.horizontalAlignment = GridData.FILL;
400 		span2.grabExcessHorizontalSpace = true;
401 		span2.heightHint = 70;
402 		fPropertyList.getTable().setLayoutData(span2);
403 
404 		fValue = new Text(instanceConfigurationGroup, SWT.WRAP);
405 		span2 = new GridData();
406 		span2.horizontalAlignment = GridData.FILL;
407 		span2.grabExcessHorizontalSpace = true;
408 		fValue.setLayoutData(span2);
409 		
410 		fValue.addFocusListener(new FocusListener() {
411 
412 			@Override
413 			public void focusLost(FocusEvent e) {
414 				// store value, once focus is lost (different element is
415 				// selected)
416 				String valueStr = fValue.getText();
417 				Slot valueSlot = null;
418 				if (m_instance == null) {
419 					// no instance has been selected
420 					// (field should be inactive, if not selected)
421 					return;
422 				}
423 				for (Slot slot : m_instance.getSlots()) {
424 					if (slot.getDefiningFeature() == m_currentAttribute) {
425 						valueSlot = slot;
426 					}
427 				}
428 				if (valueSlot == null) {
429 					// slot does not exist yet, create
430 					valueSlot = DepCreation.createSlotForConfigProp(m_instance,
431 							m_currentAttribute);
432 				}
433 				for (ValueSpecification value : valueSlot.getValues()) {
434 					if (value instanceof LiteralString) {
435 						((LiteralString) value).setValue(valueStr);
436 					} else if (value instanceof LiteralInteger) {
437 						((LiteralInteger) value).setValue(Integer
438 								.parseInt(valueStr));
439 					} else if (value instanceof LiteralBoolean) {
440 						((LiteralBoolean) value).setValue(Boolean
441 								.parseBoolean(valueStr));
442 					}
443 				}
444 			}
445 
446 			@Override
447 			public void focusGained(FocusEvent e) {
448 			}
449 		});
450 
451 		fComboValue = new Combo(instanceConfigurationGroup, SWT.NONE);
452 		span2 = new GridData();
453 		span2.horizontalAlignment = GridData.FILL;
454 		span2.grabExcessHorizontalSpace = true;
455 		fComboValue.setLayoutData(span2);
456 		
457 		fComboValue.addSelectionListener(new SelectionAdapter() {
458 
459 			@Override
460 			public void widgetSelected(SelectionEvent event) {
461 				// store value, once focus is lost (different element is
462 				// selected)
463 				int index = fComboValue.getSelectionIndex();
464 				if (m_instance == null) {
465 					// no instance has been selected
466 					// (field should be inactive, if not selected)
467 					return;
468 				}
469 				Slot valueSlot = null;
470 				for (Slot slot : m_instance.getSlots()) {
471 					if (slot.getDefiningFeature() == m_currentAttribute) {
472 						// slot for enumerations can only be configured with instanceValues.
473 						// Destroy slots with do not contain at least one instanceValue
474 						boolean destroySlot = true;
475 						for (ValueSpecification value : slot.getValues()) {
476 							if (value instanceof InstanceValue) {
477 								destroySlot = false;
478 							}
479 						}
480 						if (destroySlot) {
481 							slot.destroy();
482 						}
483 						else {
484 							valueSlot = slot;
485 						}
486 						break;
487 					}
488 				}
489 				if (valueSlot == null) {
490 					// slot does not exist yet, create
491 					valueSlot = DepCreation.createSlotForConfigProp(m_instance,
492 							m_currentAttribute);
493 				}
494 				Enumeration enumeration = (Enumeration) m_currentAttribute.getType();
495 				for (ValueSpecification value : valueSlot.getValues()) {
496 					if (value instanceof InstanceValue) {
497 						((InstanceValue) value).setInstance(enumeration.getOwnedLiterals().get(index));
498 					}
499 				}
500 			}
501 		});
502 		
503 		fDescriptionLabel = new Label(instanceConfigurationGroup, SWT.NONE);
504 		fDescriptionLabel.setText("DescriptionUtils:");
505 		fDescriptionLabel.setLayoutData(span2);
506 
507 		span2 = new GridData();
508 		span2.horizontalAlignment = GridData.FILL;
509 		span2.grabExcessHorizontalSpace = true;
510 		span2.grabExcessVerticalSpace = true;
511 		span2.verticalAlignment = GridData.FILL;
512 		span2.heightHint = 80;
513 
514 		fDescription = new Text(instanceConfigurationGroup, SWT.NONE | SWT.WRAP
515 				| SWT.V_SCROLL | SWT.READ_ONLY);
516 		fDescription.setLayoutData(span2);
517 		// createMessageArea (ruleInfoGroup);
518 		instanceConfigurationGroup.pack();
519 		setEnabled(false);
520 	}

522 	/**
523 	 * Select a rule, i.e. update the visual representation from the rule
524 	 *
525 	 * @param rule
526 	 */
527 	protected void selectProperty(Property attribute) {
528 		m_currentAttribute = attribute;
529 		if ((attribute == null) || (fValue == null)) {
530 			setEnabled(false);
531 			return;
532 		}
533 		setEnabled(true);
534 		fDescription.setText(DescriptionUtils.getDescription(attribute, "not available"));
535 		fValueLabel.setText(getValueLabel(attribute));
536 		boolean isEnum = m_currentAttribute.getType() instanceof Enumeration;
537 		fValue.setVisible(!isEnum);
538 		fComboValue.setVisible(isEnum);
539 		if (isEnum) {
540 			Enumeration enumeration = (Enumeration) m_currentAttribute.getType();
541 			EList<EnumerationLiteral> literals = enumeration.getOwnedLiterals();
542 			String items[] = new String[literals.size()];
543 			for (int i = 0; i < literals.size(); i++) {
544 				items[i] = literals.get(i).getName();
545 			}
546 			fComboValue.setItems(items);
547 		}
548 		
549 		for (Slot slot : m_instance.getSlots()) {
550 			if (slot.getDefiningFeature() == m_currentAttribute) {
551 				for (ValueSpecification value : slot.getValues()) {
552 					if (value instanceof LiteralInteger) {
553 						Integer intVal = ((LiteralInteger) value).getValue();
554 						fValue.setText(intVal.toString());
555 						return;
556 					}
557 					if (value instanceof InstanceValue) {
558 						InstanceSpecification enumIS = ((InstanceValue) value).getInstance();
559 						if (enumIS instanceof EnumerationLiteral) {
560 							EnumerationLiteral literal = (EnumerationLiteral) enumIS;
561 							int index = literal.getEnumeration().getOwnedLiterals().indexOf(literal);
562 							if (index != -1) {
563 								fComboValue.select(index);
564 							}
565 						}
566 						return;
567 					}
568 					else if (value instanceof LiteralBoolean) {
569 						Boolean boolVal = ((LiteralBoolean) value).booleanValue();
570 						fValue.setText(boolVal.toString());
571 						return;
572 					}
573 					else if (value instanceof LiteralString) {
574 						fValue.setText(((LiteralString) value).getValue());
575 						return;
576 					}
577 				}
578 			}
579 		}
580 		fValue.setText(""); //$NON-NLS-1$
581 	}

583 	private void setEnabled(boolean enabled) {
584 		fDescriptionLabel.setEnabled(enabled);
585 		fDescription.setEnabled(enabled);
586 		fValueLabel.setEnabled(enabled);
587 		fValue.setEnabled(enabled);
588 		if (!enabled) {
589 			fValue.setText(""); //$NON-NLS-1$
590 			fDescription.setText(""); //$NON-NLS-1$
591 			fValueLabel.setText(valueLabelPrefix);
592 		}
593 	}

595 	private String getValueLabel(Property attribute) {
596 		String label = valueLabelPrefix;
597 		if (attribute.getDefault() != null) {
598 			return label + " " + String.format("(default = %s)", attribute.getDefault()); //$NON-NLS-1$ //$NON-NLS-2$
599 		}
600 		return label;
601 	}
