66  	public static final String LANGUAGE_NAME = "C++"; //$NON-NLS-1$

74  	 *            the project in which the generated code should be placed

77  		this(new ProjectBasedFileAccess(project), null);

82  	 * Constructor, allows for non-standard commentHeader

88  	 */
89  	public CppModelElementsCreator(IProject project, String commentHeader) {
90  		this(new ProjectBasedFileAccess(project), commentHeader);
91  		this.project = project;

94  	/**
95  	 * Constructor. Pass caller defined file system access and commentHeader
96  	 * 
97  	 * @param project
98  	 *            the project in which the generated code should be placed
99  	 * @param commentHeader
100 	 *            commentHeader. If null, take from preferences
101 	 */
102 	public CppModelElementsCreator(IPFileSystemAccess fileSystemAccess, String commentHeader) {
103 		super(fileSystemAccess, new CppLocationStrategy(), LANGUAGE_NAME);
104 		this.commentHeader = (commentHeader != null) ?
105 				commentHeader :
106 				CppCodeGenUtils.getCommentHeader();
107 		hppExt = CppCodeGenUtils.getHeaderSuffix();
108 		cppExt = CppCodeGenUtils.getBodySuffix();
109 		sourceFolder = null;
110 	}

112 	protected String hppExt;
113 
114 	protected String cppExt;
115 
116 	protected String commentHeader;
117 
118 	/**
119 	 * Creates the files corresponding to the class. For a "simple" class
120 	 * generates 2 headers (one for the privates concrete operations and one for
121 	 * the attributes, public operations and virtual / abstract operations and
122 	 * one body file.
123 	 * 
124 	 * @param folder
125 	 * @param classifier
126 	 * @throws CoreException
127 	 */
128 	@Override
129 	protected void createPackageableElementFile(PackageableElement element, IProgressMonitor monitor) {
130 		if (sourceFolder == null) {
131 			sourceFolder = GenUtils.getSourceFolder(element);
132 		}
133 
134 		if (element instanceof Package) {
135 			generatePkg((Package) element);
136 		}
137 
138 		else if ((element instanceof PrimitiveType) || (element instanceof Enumeration) || (element instanceof Usage)) {
139 			// do nothing, included in package
140 		}
141 		else if (element instanceof Classifier) {
142 			generateClassifier((Classifier) element);
143 		}
144 		else if (element instanceof Relationship) {
145 			// no code generation for relationships
146 		}
147 		else if (element instanceof Signal) {
148 			// TODO: not supported, but do nothing
149 		}
150 		else if (element instanceof SignalEvent) {
151 			// TODO: not supported, but do nothing
152 		}
153 		else {
154 			Activator.log.debug("C++ code generator: unsupported model element " + element); //$NON-NLS-1$
155 		}
156 	}
157 
158 	protected void generateClassifier(Classifier classifier) {
159 			
160 		// treat case of manual code generation
161 		if (GenUtils.hasStereotype(classifier, ManualGeneration.class)) {
162 			final ManualGeneration mg = UMLUtil.getStereotypeApplication(classifier, ManualGeneration.class);
163 			final Include cppInclude = UMLUtil.getStereotypeApplication(classifier, Include.class);
164 			final String fileContentH = commentHeader + cppInclude.getHeader();
165 
166 			// generate header code
167 			final String fileNameH = locStrategy.getFileName(classifier) + Constants.DOT + hppExt;
168 
169 			generateFile(fileNameH, fileContentH);
170 
171 			// generate body code
172 			String fileContentB = commentHeader +
173 					cppInclude.getPreBody() + GenUtils.NL +
174 					cppInclude.getBody() + GenUtils.NL;
175 			String ext = GenUtils.maskNull(mg.getExtensionBody());
176 			if (ext.length() == 0) {
177 				ext = cppExt;
178 			}
179 			String fileNameB = locStrategy.getFileName(classifier) + Constants.DOT + ext;
180 			generateFile(fileNameB, fileContentB);
181 		}
182 
183 		// Only generate when no CppNoCodeGen stereotype is applied to the class
184 		else if ((!noCodeGen(classifier)) && (!GenUtils.hasStereotype(classifier, Template.class)) &&
185 				(!(classifier instanceof Association))) {
186 
187 			// Template Bound Class
188 			if (GenUtils.isTemplateBoundElement(classifier)) {
189 				final String bindHeaderFileName = locStrategy.getFileName(classifier) + Constants.DOT + hppExt;
190 				generateFile(bindHeaderFileName, commentHeader + CppClassifierGenerator.generateBindHeaderCode(classifier));
191 
192 				final String bindBodyFileName = locStrategy.getFileName(classifier) + Constants.DOT + cppExt;
193 				generateFile(bindBodyFileName, commentHeader + CppClassifierGenerator.generateBindBodyCode(classifier));
194 			}
195 			else {
196 				// Class Header file generation
197 				final String classHeaderFileName = locStrategy.getFileName(classifier) + Constants.DOT + hppExt;
198 				generateFile(classHeaderFileName, commentHeader + CppClassifierGenerator.generateClassHeaderCode(classifier));
199 
200 				// Class Body file generation
201 				if (classifier instanceof Class) {
202 					final String classBodyFileName = locStrategy.getFileName(classifier) + Constants.DOT + cppExt;
203 					generateFile(classBodyFileName, commentHeader + CppClassifierGenerator.generateClassBodyCode(classifier));
204 				}
205 			}
206 		}
207 	}
208 
209 	protected void generatePkg(Package pkg) {
210 		final String fileName = locStrategy.getFileName(pkg) + Constants.DOT + hppExt;
211 		generateFile(fileName, CppPackageHeaderGenerator.generateCode(pkg).toString());
212 	}
213 
214 	protected void generateFile(String fileName, String content) {
215 		fileSystemAccess.generateFile(sourceFolder + fileName, format(content));
216 	}
217 
218 	/**
219 	 * Apply the user's currently selected formatting options to the input content. Return the
220 	 * input String in case of error.
221 	 */
222 	private static String format(String content) {
223 
224 		// do nothing if the CDT plugin is not loaded
225 		if (Platform.getBundle(CCorePlugin.PLUGIN_ID) == null)
226 			return content;
227 
228 		CodeFormatter codeFormatter = ToolFactory.createCodeFormatter(null);
229 		IDocument doc = new Document(content);
230 
231 		TextEdit edit = codeFormatter.format(CodeFormatter.K_TRANSLATION_UNIT, doc.get(), 0, doc.get().length(), 0, null);
232 
233 		if (edit == null) {
234 			Activator.log.debug(Messages.CppModelElementsCreator_CannotFormatContent);
235 			return content;
236 		}
237 
238 		try {
239 			edit.apply(doc);
240 			return doc.get();
241 		} catch (MalformedTreeException e) {
242 			Activator.log.error(e);
243 		} catch (BadLocationException e) {
244 			Activator.log.error(e);
245 		}
246 		return content;
247 	}
248 
249 	protected boolean noCodeGen(Element element) {
250 		if (GenUtils.hasStereotype(element, NoCodeGen.class) ||
251 				GenUtils.hasStereotype(element, External.class) ||
252 				GenUtils.hasStereotypeTree(element, ExternLibrary.class)) {
253 			return true;
254 		}
255 		return super.noCodeGen(element);
256 	}
