41  /**
42   * Flatten interaction components for distributed communication, i.e. remove the enclosing interaction component
43   * and only deploy the fragments.
44   * This is required, since the interaction component itself cannot be deployed on several nodes.
45   */
46  public class FlattenInteractionComponents implements IM2MTrafoElem {
47  
48  	private static final String FLATTEN_SEP = "_"; //$NON-NLS-1$
49  
50  	/**
51  	 *
52  	 * Flatten a composite interaction component, i.e. replace it by the containing fragments. This includes the following actions
53  	 * 1. Parts typed with the interaction component must be replaced with a set of parts typed with the fragment.
54  	 * 2. Connectors must be redirected.
55  	 *
56  	 * Please note that we do not delete the original interaction component, but it will not appear in a model generated for a specific node.
57  	 * TODO: We need to re-target connectors in all containing composites, but the same containingComposite might have several instances within an application. In this case,
58  	 * we only need to move slots.
59  	 *
60  	 * Well, there are different classes: first: the flattened interaction component and other components that reference the flattened interaction component.
61  	 *
62  	 * @param composite
63  	 *            a composite class
64  	 * @param instance
65  	 *            the instance associated with the composite class (1st parameter)
66  	 * @param containingInstance
67  	 *            an instance of the containing composite
68  	 * @param containingSlot
69  	 *            the slot associated with the instance (2nd parameter)
70  	 */
71  	public void flattenAssembly(Class composite, InstanceSpecification instance, InstanceSpecification containingInstance, Slot containingSlot)
72  	{
73  		Classifier containingCompositeCl = DepUtils.getClassifier(containingInstance);
74  		StructuralFeature sfForIA = containingSlot.getDefiningFeature();
75  		Map<Property, Property> replaceParts = new HashMap<Property, Property>();
76  		if ((containingCompositeCl instanceof Class) && (sfForIA instanceof Property)) {
77  			Class containingComposite = (Class) containingCompositeCl;
78  			Property partForIA = (Property) sfForIA;
79  
80  			for (Property fragmentPart : composite.getAllAttributes()) {
81  				if (fragmentPart instanceof Port) {
82  					continue;
83  				}
84  				String partName = partForIA.getName() + FLATTEN_SEP + fragmentPart.getName();
85  				// create a new part in the containing composite
86  				Property newPartForFragment = containingComposite.createOwnedAttribute(partName, fragmentPart.getType());
87  				replaceParts.put(fragmentPart, newPartForFragment);
88  
89  				boolean foundConnector = false;
90  				// now redirect connectors: find whether a port delegates to the fragment
91  				for (Connector connector : composite.getOwnedConnectors()) {
92  					if (ConnectorUtil.connectsPart(connector, fragmentPart)) {
93  						foundConnector = true;
94  						// internal connector for the part, check whether delegation
95  						ConnectorEnd ce = ConnectorUtil.connEndNotPart(connector, fragmentPart);
96  						if ((ce != null) && (ce.getPartWithPort() == null)) {
97  							// delegation connector, need to re-targed connections to external port
98  							Port port = (Port) ce.getRole();
99  							// now look for connections to this port in the containingComposite and shortcut these ...
100 							retargetConnections(containingComposite, port, partForIA, newPartForFragment);
101 						} else {
102 							// assembly: assembly connection between fragments are not authorized
103 						}
104 					}
105 				}
106 				if (!foundConnector) {
107 					throw new RuntimeException("the part <" + fragmentPart.getName() + "> within composite <" + composite.getName() + "> is not connected with any of its ports"); //$NON-NLS-1$ //$NON-NLS-2$ //$NON-NLS-3$
108 				}
109 			}
110 			// on instance level: move slots from instance to containingInstance
111 			EList<Slot> slots = instance.getSlots();
112 			containingInstance.getSlots().addAll(slots);
113 			// replace defining feature
114 			for (Slot slot : containingInstance.getSlots()) {
115 				StructuralFeature sf = slot.getDefiningFeature();
116 				if (replaceParts.containsKey(sf)) {
117 					slot.setDefiningFeature(replaceParts.get(sf));
118 					// assure naming convention for instances: here, we update the name of the instance
119 					// and its sub-instances with the name of the containingInstance
120 					InstanceSpecification subInstance = DepUtils.getInstance(slot);
121 					if ((subInstance != null) && !DepUtils.isShared(slot)) {
122 						DepPlanUtils.updateInstanceNames(subInstance, instance.getName() + FLATTEN_SEP + sf.getName());
123 					}
124 				}
125 			}
126 
127 			containingSlot.destroy();
128 			// destroy relationships of flattened instance, in particular allocation
129 			for (DirectedRelationship dr : instance.getSourceDirectedRelationships()) {
130 				dr.destroy();
131 			}
132 			instance.destroy();
133 			partForIA.destroy();
134 
135 			// move is
136 			/*
137 			 * InstanceSpecification partIS;
138 			 * DepUtils.createSlot (cdp, newIS, partIS, newPart);
139 			 */
140 		}
141 	}
142 
143 	/**
144 	 * Find a connection within a connecting
145 	 *
146 	 * @param containingComposite
147 	 * @param port
148 	 *            A port which delegates to a fragment
149 	 * @param part
150 	 *            the part within the containing composite (1st parameter) point to the interactionComponent
151 	 * @param subPart
152 	 *            a newly created part corresponding to the fragment
153 	 */
154 	public void retargetConnections(Class containingComposite, Port port, Property part, Property subPart) {
155 		for (Connector connector : containingComposite.getOwnedConnectors()) {
156 			if (ConnectorUtil.connectsPart(connector, part) && ConnectorUtil.connectsPort(connector, port)) {
157 				ConnectorEnd ce = ConnectorUtil.connEndForPart(connector, part);
158 				if (ce != null) {
159 					// TODO: only with with connections targeting a port of a part, not with
160 					// those targeting directly a part
161 					ce.setPartWithPort(subPart);
162 				}
163 			}
164 		}
165 	}
166 
167 	@Override
168 	public void transformElement(M2MTrafo trafo, Element element) throws TransformationException {
169 		// TODO Auto-generated method stub
170 		if (element instanceof InstanceSpecification) {
171 			InstanceSpecification instance = (InstanceSpecification) element;
172 			Classifier cl = DepUtils.getClassifier(instance);
173 			if (cl instanceof Class) {
174 				InteractionComponent ic = UMLUtil.getStereotypeApplication(cl, InteractionComponent.class);
175 				if ((ic != null) && ic.isForDistribution()) {
176 					Slot containingSlot = DepUtils.getParentSlot(instance);
177 					InstanceSpecification containingInstance = containingSlot.getOwningInstance();
178 					flattenAssembly((Class) cl, instance, containingInstance, containingSlot);
179 				}
180 			}
181 		}
182 	}
