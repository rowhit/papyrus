52  	/**
53  	 * Get the template signature of a classifier. If the classifier does not
54  	 * have a signature, examine if the owning class provides one
55  	 *
56  	 * @param template
57  	 *            The potential template
58  	 * @return the signature or null, if none can be found.
59  	 */
60  	public static TemplateSignature getSignature(TemplateableElement template) {
61  		TemplateSignature signature = getSignatureDirect(template);
62  		if (signature != null) {
63  			return signature;
64  		}
65  
66  		// no signature found, try signature of owning classifier (nested class)
67  		// or package
68  		Element owner = template.getOwner();
69  		if (owner instanceof TemplateableElement) {
70  			// owned by a classifier, not a package
71  			return getSignature((TemplateableElement) owner);
72  		} else {
73  			return null;
74  		}
75  	}
76  
77  	public static Element getTemplateOwner(Element ne, TemplateSignature signature) {
78  		if ((ne instanceof TemplateableElement) && getSignatureDirect((TemplateableElement) ne) == signature) {
79  			return ne;
80  		}
81  		Element owner = ne.getOwner();
82  		if (owner != null) {
83  			return getTemplateOwner(owner, signature);
84  		}
85  		return null;
86  	}

88  	/**
89  	 * Get the template signature of a templateable element (typically a package). The class must
90  	 * (1) either own the signature
91  	 * (2) or merge with a package which owns the signature.
92  	 * Qompass enables the "extension" of existing packages via the package merge mechanism
93  	 *
94  	 * @param template
95  	 *            The potential template
96  	 * @return the signature or null, if none can be found.
97  	 */
98  
99  	public static TemplateSignature getSignatureDirect(TemplateableElement template) {
100 		for (Element element : template.getOwnedElements()) {
101 			if (element instanceof TemplateSignature) {
102 				return (TemplateSignature) element;
103 			}
104 		}
105 
106 		// enable multiple package templates sharing the same signature.
107 		if (template instanceof Package) {
108 			Package pkg = (Package) template;
109 			for (PackageMerge pkgImport : pkg.getPackageMerges()) {
110 				Package importedPkg = pkgImport.getMergedPackage();
111 				return getSignature(importedPkg);
112 			}
113 		}
114 		return null;
115 	}

117 	/**
118 	 * Create a template binding by using a single, fixed actual (used within
119 	 * Qompass for the binding of container extensions depending on the component
120 	 * executor)
121 	 *
122 	 * @param model
123 	 *            : target model in which to create the bound package
124 	 * @param template
125 	 * @param fixedActual
126 	 * @return
127 	 * @throws TransformationException
128 	 */
129 	public static TemplateBinding fixedBinding(Package model, TemplateableElement template, Classifier fixedActual)
130 			throws TransformationException {
131 		// obtain the signature of an element within a package template.
132 
133 		TemplateSignature signature = getSignature(template);
134 		if (signature == null) {
135 			// not a template, retain original name
136 			Activator.log.info(String.format(
137 					Messages.TemplateUtils_NoTemplateSignature, (template instanceof NamedElement ?
138 							((NamedElement) template).getName() : "undef"))); //$NON-NLS-1$
139 			return null;
140 		}
141 
142 		EList<Type> actuals = new BasicEList<Type>();
143 
144 		Package pkgTemplate = (Package) signature.getOwner();
145 		String name = pkgTemplate.getName();
146 
147 		// loop on template parameters;
148 		// for (TemplateParameter parameter : signature.getOwnedParameters ()) {
149 		// ParameterableElement formal = parameter.getParameteredElement();
150 
151 		// now obtain suitable binding for this parameter - look for ports that
152 		// are typed with the formal template parameter
153 
154 		actuals.add(fixedActual);
155 		name = name + "_" + fixedActual.getName(); //$NON-NLS-1$
156 		// }
157 
158 		// the bound package is instantiated in the same model, in which the
159 		// composite can be found (avoid modifying an imported model).
160 		// todo: root model as an additional parameter?
161 		Namespace owner = (Namespace) pkgTemplate.getOwner();
162 		owner = CreationUtils.getAndCreate(model, owner.allNamespaces());
163 
164 		Package boundPackage = (Package) owner.getMember(name);
165 		if (boundPackage == null) {
166 			// class does not exist yet, needs to be created.
167 			boundPackage = ((Package) owner).createNestedPackage(name);
168 
169 			Activator.log.info(String.format(
170 					Messages.TemplateUtils_InfoCreateBoundPackage, name, owner.getName()));
171 		}
172 
173 		TemplateBinding binding = boundPackage.getTemplateBinding(signature);
174 		if (binding == null) {
175 			// binding is not existing yet (should normally only happen, if the class has
176 			// just been created - but it's better to re-check, even if the bound package
177 			// was already there)
178 			binding = boundPackage.createTemplateBinding(signature);
179 
180 			Iterator<Type> actualsIter = actuals.iterator();
181 
182 			// loop on template parameters;
183 			for (TemplateParameter parameter : signature.getOwnedParameters()) {
184 
185 				TemplateParameterSubstitution substitution =
186 						binding.createParameterSubstitution();
187 				substitution.setFormal(parameter);
188 
189 				// now obtain suitable binding for this parameter - look for
190 				// ports that are typed with
191 				// the formal template parameter
192 				Type actual = actualsIter.next();
193 				substitution.setActual(actual);
194 			}
195 		}
196 		return binding;
197 	}

200 	 * create a "sub" binding in which the first parameter of type Cl is assumed
201 	 * to bind the sub-signature TODO: compared type and name of parameters in
202 	 * two signatures TODO: support more than one template parameter
203 	 *
204 	 * @param model
205 	 * @param te
206 	 * @param existingBinding
207 	 * @return
208 	 * @throws TransformationException
209 	 */
210 	public static TemplateBinding getSubBinding(Package model,
211 			TemplateableElement te, TemplateBinding existingBinding)
212 			throws TransformationException {
213 
214 		for (TemplateParameterSubstitution tps : existingBinding.getParameterSubstitutions()) {
215 			ParameterableElement pe = tps.getActual();
216 			if (pe instanceof Classifier) {
217 				return fixedBinding(model, te, (Classifier) pe);
218 			}
219 		}
220 		return null;
221 	}

237 	 * Return the actual for a potential formal parameter within a
238 	 * templateBinding (move to TemplateUtils?)
239 	 *
240 	 * @param binding
241 	 *            a template binding
242 	 * @param formal
243 	 *            a potential formal parameter, i.e. a parameter for which we
244 	 *            check, if it really corresponds to a formal parameter within
245 	 *            the template binding.
246 	 * @return the actual parameter that is associated with the potential formal
247 	 *         parameter, or null if the 2nd parameter does not correspond to a
248 	 *         formal parameter of the binding.
249 	 */
250 	public static Classifier getActualFromBinding(TemplateBinding binding, Type formal) {
251 		for (TemplateParameterSubstitution substitution : binding.getParameterSubstitutions()) {
252 			ParameterableElement pe = substitution.getFormal().getParameteredElement();
253 			if (pe == formal) {
254 				Activator.log.info(String.format(
255 						Messages.TemplateUtils_InfoGetActualFrom, pe));
256 				return (Classifier) substitution.getActual();
257 			}
258 		}
259 		return null;
260 	}

262 	public static Classifier getActualFromBinding(TemplateBinding binding, String formalName) {
263 		for (TemplateParameterSubstitution substitution : binding.getParameterSubstitutions()) {
264 			ParameterableElement pe = substitution.getFormal().getParameteredElement();
265 			Activator.log.info(String.format(
266 					Messages.TemplateUtils_InfoGetActualFrom, pe));
267 			if ((pe instanceof NamedElement)
268 					&& ((NamedElement) pe).getName().equals(formalName)) {
269 				return (Classifier) substitution.getActual();
270 			}
271 		}
272 		return null;
273 	}

292 	/**
293 	 * Return a sequence of namespaces for a given element, starting from the "bottom"
294 	 * one, i.e. the one in which the element is contained. It will end before the
295 	 * searchNS namespace is reached. Returns null, if the element is not contained
296 	 * within the search namespace.
297 	 * This function will put a merged package into the path (instead of the owner), enabling
298 	 * the extension of existing packages.
299 	 *
300 	 * @param element
301 	 * @param searchNS
302 	 * @return
303 	 */
304 	public static EList<Namespace> relativePathWithMerge(Element element, Namespace searchNS) {
305 		EList<Namespace> pathList = new BasicEList<Namespace>();
306 		Element owner = element.getOwner();
307 		if (!(owner instanceof Namespace)) {
308 			// happens, if element is contained in a template signature
309 			return null;
310 		}
311 		Namespace ns = (Namespace) owner;
312 		while (ns != null) {
313 			if (ns == searchNS) {
314 				return pathList;
315 			}
316 			pathList.add(ns);
317 
318 			if (ns instanceof Package) {
319 				Package pkg = (Package) ns;
320 				Iterator<PackageMerge> pkgMerges = pkg.getPackageMerges().iterator();
321 				// if package merge exists, get first merged package and add it to path
322 				if (pkgMerges.hasNext()) {
323 					PackageMerge pkgImport = pkgMerges.next();
324 					ns = pkgImport.getMergedPackage();
325 					continue;
326 				}
327 			}
328 
329 			ns = (Namespace) ns.getOwner();
330 		}
331 		return null;
332 	}

334 	/**
335 	 * Re-target connectors after a part has changed its type from template to an instantiation
336 	 * of this template. In this case, the roles of the connector ends still reference the port
337 	 * of the template instead of the bound template binding.
338 	 * The new roles are assigned based on an equal name, assuming that template instantiation
339 	 * does not change the name of the ports.
340 	 *
341 	 * @param composite
342 	 *            a composite containing connectors
343 	 * @param part
344 	 *            a part within the composite whose type has changed.
345 	 */
346 	public static void retargetConnectors(StructuredClassifier composite, Property part) {
347 		Type partType = part.getType();
348 		if (partType instanceof EncapsulatedClassifier) {
349 			EncapsulatedClassifier partTypeEC = (EncapsulatedClassifier) partType;
350 			for (Connector connector : composite.getOwnedConnectors()) {
351 				if (ConnectorUtil.connectsPart(connector, part)) {
352 					// the connector end targets a port of a part or the composite (in case of delegation)
353 					ConnectorEnd connEnd = ConnectorUtil.connEndForPart(connector, part);
354 					// redirect role, if pointing to port
355 					if (connEnd.getRole() instanceof Port) {
356 						Port connectedTemplatePort = (Port) connEnd.getRole();
357 						Port connectedBoundPort = (Port) ElementUtils.getNamedElementFromList(
358 								PortUtils.getAllPorts(partTypeEC), connectedTemplatePort.getName());
359 						connEnd.setRole(connectedBoundPort);
360 					}
361 				}
362 			}
363 		}
364 	}

366 	/**
367 	 * Re-target connectors after an unknown number of parts have changed their type from template
368 	 * to an instantiation of this template. In this case, the roles of the connector ends still
369 	 * reference the port of the template instead of the bound template binding.
370 	 * The new roles are assigned based on an equal name, assuming that template instantiation
371 	 * does not change the name of the ports.
372 	 *
373 	 * @param composite
374 	 *            a composite containing connectors
375 	 */
376 	public static void retargetConnectors(StructuredClassifier composite) {
377 		for (Connector connector : composite.getOwnedConnectors()) {
378 			// the connector end targets a port of a part or the composite (in case of delegation)
379 			for (ConnectorEnd connEnd : connector.getEnds()) {
380 				Property part = connEnd.getPartWithPort();
381 				if ((part != null) && (part.getType() instanceof EncapsulatedClassifier)) {
382 					EncapsulatedClassifier partTypeEC = (EncapsulatedClassifier) part.getType();
383 					ConnectableElement role = connEnd.getRole();
384 					EList<Port> ports = PortUtils.getAllPorts(partTypeEC);
385 					if ((role instanceof Port) && !ports.contains(role)) {
386 						// role is not in list of ports
387 						Port connectedBoundPort = (Port) ElementUtils.getNamedElementFromList(ports, role.getName());
388 						connEnd.setRole(connectedBoundPort);
389 					}
390 				}
391 			}
392 		}
