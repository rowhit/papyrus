11  package org.eclipse.papyrus.designer.transformation.core.templates;
12  
13  import java.lang.reflect.InvocationTargetException;
14  import java.lang.reflect.Method;
15  
16  import org.eclipse.papyrus.designer.transformation.base.utils.TransformationException;
17  import org.eclipse.papyrus.designer.transformation.core.Messages;
18  import org.eclipse.papyrus.designer.transformation.extensions.ITextTemplate;
19  import org.eclipse.papyrus.designer.transformation.extensions.TextTemplateExt;
20  import org.eclipse.uml2.uml.Element;
21  import org.eclipse.uml2.uml.NamedElement;
22  
23  /**
24   * Support for binding/instantiation of text templates
25   */
26  public class TextTemplateBinding {
27  
28  	public static final String TEMPLATE_MAGIC = "!template"; //$NON-NLS-1$
29  
30  	/**
31  	 * Simple template expression that is replaced with the name of a named
32  	 * element. Uses an Acceleo syntax
33  	 * The objective is that the access to an element name is very common and does
34  	 * not need an xtend script behind.
35  	 */
36  	private static final String NAME_IN_BRACKETS = "[name/]"; //$NON-NLS-1$
37  
38  	/**
39  	 * Simple template expression that is replaced with the name of a named
40  	 * element. Slight variation of the Acceleo syntax
41  	 */
42  	private static final String NAME_IN_BRACKETS_SIMPLE = "[name]"; //$NON-NLS-1$
43  
44  	/**
45  	 * Execute a script passed as parameter. It will be executed in the context
46  	 * of a base script that declares the meta-model (UML2) and has additional
47  	 * imports (currently none).
48  	 *
49  	 * @param template
50  	 *            The template in string form
51  	 * @param element
52  	 *            The UML element to which the template is applied
53  	 * @return the evaluated template
54  	 */
55  	public static String bind(String templateStr, Element element) throws TransformationException {
56  		return bind(templateStr, element, null);
57  	}
58  
59  	/**
60  	 * Execute a script passed as parameter. It will be executed in the context
61  	 * of a base script that declares the meta-model (UML2). It allows for
62  	 * additional arguments
63  	 *
64  	 * @param template
65  	 *            The template in string form
66  	 * @param element
67  	 *            The UML element to which the template is applied
68  	 * @return the evaluated template
69  	 */
70  	public static String bind(String templateStr, Element element, Object args[]) throws TransformationException {
71  		if (templateStr.startsWith(TEMPLATE_MAGIC)) {
72  			return bindTemplate(templateStr, element, args);
73  		}
74  		else if (element instanceof NamedElement) {
75  			// simple replacement of specific template entry [name]
76  			String name = ((NamedElement) element).getName();
77  			if (templateStr.contains(NAME_IN_BRACKETS)) {
78  				return templateStr.replace(NAME_IN_BRACKETS, name);
79  			}
80  			else if (templateStr.contains(NAME_IN_BRACKETS_SIMPLE)) {
81  				return templateStr.replace(NAME_IN_BRACKETS_SIMPLE, name);
82  			}
83  		}
84  		return templateStr;
85  	}
86  
87  	/**
88  	 * Execute a script passed as parameter. It will be executed in the context
89  	 * of a base script that declares the meta-model (UML2) and has additional
90  	 * imports (currently none).
91  	 *
92  	 * @param template
93  	 *            The template in string form
94  	 * @param element
95  	 *            The UML element to which the template is applied
96  	 * @return the evaluated template
97  	 */
98  	public static String bindTemplate(String templateStr, Element element, Object args[]) throws TransformationException {
99  		String[] templateRef = templateStr.substring(TEMPLATE_MAGIC.length() + 1).split("\\."); //$NON-NLS-1$
100 		String templateId = templateRef[0].trim();
101 		String methodName = templateRef[1].trim();
102 		ITextTemplate generator = TextTemplateExt.getTextTemplate(templateId);
103 		if (generator == null) {
104 			throw new TransformationException(String.format(Messages.TextTemplateBinding_TemplateNotFound, templateId, methodName));
105 		}
106 		
107 		try {
108 			Object result;
109 			if (args != null) {
110 				Method method = getMethod(generator, methodName, 1+args.length);
111 				result = method.invoke(generator, element, args);
112 			}
113 			else {
114 				Method method = getMethod(generator, methodName, 1);
115 				result = method.invoke(generator, element);
116 			}
117 			if (result instanceof String) {
118 				return (String) result;
119 			}
120 			else if (result instanceof CharSequence) {
121 				return result.toString();
122 			}
123 			else {
124 				throw new TransformationException(Messages.TextTemplateBinding_TemplateResultIsNotAString);
125 			}
126 		} catch (SecurityException | IllegalAccessException
127 				| IllegalArgumentException | InvocationTargetException | NullPointerException e) {
128 			throw new RuntimeException(e);
129 		}
130 	}
131 
132 	public static Method getMethod(ITextTemplate generator, String methodName, int numberOfArguments) {
133 		for (Method method : generator.getClass().getMethods()) {
134 			if (method.getName().equals(methodName) && method.getParameterTypes().length == numberOfArguments) {
135 				return method;
136 			}
137 		}
138 		return null;
139 	}
140 }
