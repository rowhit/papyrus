31  /**
32   * Handler for C++ code generation
33   */
34  public class GenerateCodeHandler extends CmdHandler {
35  
36  	private static final boolean Headless = Boolean.getBoolean("papyrus.run-headless");
37  
38  	// ------------------------------------------------------------------------
39  	// Execution
40  	// ------------------------------------------------------------------------
41  
42  	@Override
43  	public boolean isEnabled() {
44  		updateSelectedEObject();
45  
46  		if (selectedEObject instanceof Package || selectedEObject instanceof Classifier) {
47  			URI uri = selectedEObject.eResource().getURI();
48  
49  			// URIConverter uriConverter = resource.getResourceSet().getURIConverter();
50  			IWorkspaceRoot root = ResourcesPlugin.getWorkspace().getRoot();
51  			if (uri.segmentCount() < 2) {
52  				return false;
53  			}
54  			IProject modelProject = root.getProject(uri.segment(1));
55  			return modelProject.exists();
56  		}
57  
58  		return false;
59  	}
60  
61  	/**
62  	 * Generate code for a passed packageable element and the elements required by it (in the sense of #included statements)
63  	 *
64  	 * @param mec
65  	 *            model elements creator
66  	 * @param pe
67  	 *            the element that should be generated
68  	 * @param alreadyHandled
69  	 *            list of packageable elements for which code has already been generated.
70  	 * @param recurse
71  	 *            if the passed packageableElement is a package, generate code for its contents (recursively).
72  	 */
73  	public void generate(CppModelElementsCreator mec, PackageableElement pe, EList<PackageableElement> alreadyHandled, boolean recurse) {
74  		alreadyHandled.add(pe);
75  		mec.createPackageableElement(pe, null, false);
76  		// Eventual refresh is done in createPackageableElement
77  
78  		// if recursion is active, go into all sub-elements
79  		if (pe instanceof Package && recurse) {
80  			for (PackageableElement subPe : ((Package) pe).getPackagedElements()) {
81  				if (!alreadyHandled.contains(subPe)) {
82  					generate(mec, subPe, alreadyHandled, recurse);
83  				}
84  			}
85  		}
86  
87  		// add required classifiers
88  		if (pe instanceof Classifier) {
89  			EList<Classifier> requiredClassifiers = ClassUtils.requiredClassifiers((Classifier) pe);
90  			
91  			for (Classifier requiredClassifier : requiredClassifiers) {
92  				if (!alreadyHandled.contains(requiredClassifier)) {
93  					generate(mec, requiredClassifier, alreadyHandled, false);
94  				}
95  			}
96  		}
97  		
98  		// owning package is required by generated code.
99  		Package owningPackage = pe.getNearestPackage();
100 		if ((owningPackage != null) && (owningPackage != pe)) {
101 			if (!alreadyHandled.contains(owningPackage)) {
102 				generate(mec, owningPackage, alreadyHandled, false);
103 			}
104 		}
105 	}
106 
107 	@Override
108 	public Object execute(ExecutionEvent event) throws ExecutionException {
109 
110 		if (selectedEObject instanceof PackageableElement) {
111 			PackageableElement pe = (PackageableElement) selectedEObject;
112 
113 			IProject modelProject = LocateCppProject.getTargetProject(pe, true);
114 			if (modelProject == null) {
115 				return null;
116 			}
117 
118 			// get the container for the current element
119 			CppModelElementsCreator mec = new CppModelElementsCreator(modelProject);
120 			generate(mec, pe, new BasicEList<PackageableElement>(), true);
121 		}
122 		return null;
123 	}
124 }
