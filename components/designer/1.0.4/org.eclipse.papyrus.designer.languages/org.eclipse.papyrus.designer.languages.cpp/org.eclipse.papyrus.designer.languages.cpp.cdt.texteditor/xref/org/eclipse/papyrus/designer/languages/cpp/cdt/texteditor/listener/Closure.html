15  package org.eclipse.papyrus.designer.languages.cpp.cdt.texteditor.listener;
16  
17  import org.eclipse.emf.common.util.BasicEList;
18  import org.eclipse.emf.common.util.EList;
19  import org.eclipse.emf.ecore.EObject;
20  import org.eclipse.emf.ecore.EStructuralFeature.Setting;
21  import org.eclipse.uml2.common.util.UML2Util;
22  import org.eclipse.uml2.uml.Association;
23  import org.eclipse.uml2.uml.Classifier;
24  import org.eclipse.uml2.uml.Element;
25  
26  /**
27   * Calculate the closure of classifiers that need to be re-generated, if a
28   * classifier changes
29   * 
30   * There are two different reasons for the need to re-generate a classifier:
31   * (1) one of the elements that it contains has changed
32   * (2) an element outside the containment close is changed. An example is a superclass
33   * that changes its name or an association target. In many cases, classifier depend
34   * on names of outside elements, but the also require a regeneration if the element
35   * is deleted.
36   * We need to take the change-kind into account. E.g. if the name of an operation in A changes, we need to
37   * re-generate class A, but we only need to regenerate class B (inheriting from A), if it actually
38   * redefines that operation.
39   */
40  public class Closure {
41  	public static EList<Classifier> calc(Classifier cl) {
42  		EList<Classifier> list = new BasicEList<Classifier>();
43  		calc(cl, list);
44  		return list;
45  	}
46  
47  	public static void calc(Classifier cl, EList<Classifier> list) {
48  		if (cl != null) {
49  			for (Setting setting : UML2Util.getNonNavigableInverseReferences(cl)) {
50  				EObject eObj = setting.getEObject();
51  				if (eObj instanceof Element) {
52  					Classifier referencedCl = nearestClassifier(eObj);
53  					if (!list.contains(referencedCl) && (!(referencedCl instanceof Association))) {
54  						list.add(referencedCl);
55  						calc(referencedCl, list);
56  					}
57  				}
58  			}
59  		}
60  	}
61  
62  	public static boolean isContained(EObject parent, EObject possibleChild) {
63  		while (possibleChild != null) {
64  			if (parent == possibleChild) {
65  				return true;
66  			}
67  			possibleChild = possibleChild.eContainer();
68  		}
69  		return false;
70  	}
71  
72  	public static Classifier nearestClassifier(EObject eObj) {
73  		while (eObj != null) {
74  			if (eObj instanceof Classifier) {
75  				return (Classifier) eObj;
76  			}
77  			eObj = eObj.eContainer();
78  		}
79  		return null;
80  	}
81  }
