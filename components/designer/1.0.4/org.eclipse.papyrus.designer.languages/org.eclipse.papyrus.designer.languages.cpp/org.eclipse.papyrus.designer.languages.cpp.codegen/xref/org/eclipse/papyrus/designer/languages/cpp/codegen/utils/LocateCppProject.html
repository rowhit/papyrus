33  public class LocateCppProject {
34  	private static final boolean Headless = Boolean.getBoolean("papyrus.run-headless"); //$NON-NLS-1$
35  
36  	/**
37  	 * Locate and return the target project for the given packageable element. Return null if
38  	 * no target project can be found.
39  	 *
40  	 * Ensures that the target project is correctly setup to contain generated C/C++ code. Does
41  	 * not create a new project, but may modify existing ones.
42  	 *
43  	 * @param pe
44  	 *            a packageable element within a model
45  	 * @param createIfMissing
46  	 *            if true, ask the user to apply the C++ nature if required.
47  	 * @return the associated project, if the C++ nature is applied.
48  	 */
49  	public static IProject getTargetProject(PackageableElement pe, boolean createIfMissing) {
50  		Package rootPkg = PackageUtil.getRootPackage(pe);
51  
52  		IWorkspaceRoot root = ResourcesPlugin.getWorkspace().getRoot();
53  
54  		IPreferenceStore store = Activator.getDefault().getPreferenceStore();
55  		String prefix = store != null ? store.getString(CppCodeGenConstants.P_PROJECT_PREFIX) : "bad preferences."; //$NON-NLS-1$
56  		
57  		org.eclipse.papyrus.designer.languages.common.profile.Codegen.Project projectStereo =
58  				UMLUtil.getStereotypeApplication(rootPkg, org.eclipse.papyrus.designer.languages.common.profile.Codegen.Project.class);
59  		
60  		String projectName;
61  		if (projectStereo != null) {
62  			projectName = projectStereo.getProjectName();
63  		}
64  		else {
65  			projectName = prefix + rootPkg.getName();
66  					
67  		}
68  		IProject modelProject = root.getProject(projectName);
69  		
70  		if (!modelProject.exists()) {
71  			if (Headless)
72  			{
73  				try {
74  					modelProject.create(null);
75  				} catch (CoreException e) {
76  					return null;
77  				}
78  			}
79  			else
80  			{
81  				boolean create = createIfMissing && openQuestion(
82  						Messages.LocateCppProject_CreateTargetProjectTitle,
83  						String.format(Messages.LocateCppProject_CreateTargetProjectDesc, projectName));
84  				if (create) {
85  					ILangProjectSupport langSupport = LanguageProjectSupport.getProjectSupport(CppModelElementsCreator.LANGUAGE_NAME);
86  					if (langSupport != null) {
87  						modelProject = langSupport.createProject(projectName);
88  						langSupport.setSettings(modelProject, langSupport.initialConfigurationData());
89  						if (modelProject == null) {
90  							return null;
91  						}
92  					}
93  					else {
94  						return null;
95  					}
96  				}
97  				else {
98  					return null;
99  				}
100 			}
101 		}
102 
103 		// Make sure the target project is open. If it was just created then it is likely open,
104 		// however if it is an old project then it could have been closed.
105 		if (!modelProject.isOpen()) {
106 			try {
107 				modelProject.open(null);
108 			} catch (CoreException e) {
109 				return null;
110 			}
111 		}
112 
113 		// Make sure the target project has the C and C++ build natures.
114 		try {
115 			if (!modelProject.hasNature(CCProjectNature.CC_NATURE_ID)) {
116 				boolean apply = createIfMissing && (Headless || openQuestion(
117 						Messages.LocateCppProject_ApplyCNatureTitle,
118 						Messages.LocateCppProject_ApplyCNatureDesc));
119 				if (!apply) {
120 					return null;
121 				}
122 				CProjectNature.addCNature(modelProject, null);
123 				CCProjectNature.addCCNature(modelProject, null);
124 			}
125 		} catch (CoreException e) {
126 			Activator.log.error(e);
127 		}
128 		return modelProject;
129 	}
130 	
131 	private static boolean openQuestion(final String title, final String message)
132 	{
133 		final boolean[] ret = new boolean[] { false };
134 		Display.getDefault().syncExec(new Runnable() {
135 			@Override
136 			public void run() {
137 				ret[0] = MessageDialog.openQuestion(Display.getCurrent().getActiveShell(), title, message);
138 			}
139 		});
140 		return ret[0];
141 	}
