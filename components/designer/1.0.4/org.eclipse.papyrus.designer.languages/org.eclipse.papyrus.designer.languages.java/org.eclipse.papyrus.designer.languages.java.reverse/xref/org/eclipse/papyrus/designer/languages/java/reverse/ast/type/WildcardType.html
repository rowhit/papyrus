30  package org.eclipse.papyrus.designer.languages.java.reverse.ast.type;
31  
32  import org.eclipse.papyrus.designer.languages.java.reverse.ast.visitor.GenericVisitor;
33  import org.eclipse.papyrus.designer.languages.java.reverse.ast.visitor.VoidVisitor;
34  
35  /**
36   * @author Julio Vilmar Gesser
37   */
38  public final class WildcardType extends Type {
39  
40  	private ReferenceType ext;
41  
42  	private ReferenceType sup;
43  
44  	public WildcardType() {
45  	}
46  
47  	public WildcardType(int beginLine, int beginColumn, int endLine, int endColumn, ReferenceType ext, ReferenceType sup) {
48  		super(beginLine, beginColumn, endLine, endColumn);
49  		assert ext == null || sup == null;
50  		this.ext = ext;
51  		this.sup = sup;
52  	}
53  
54  	@Override
55  	public <R, A> R accept(GenericVisitor<R, A> v, A arg) {
56  		return v.visit(this, arg);
57  	}
58  
59  	@Override
60  	public <A> void accept(VoidVisitor<A> v, A arg) {
61  		v.visit(this, arg);
62  	}
63  
64  	public ReferenceType getExtends() {
65  		return ext;
66  	}
67  
68  	public ReferenceType getSuper() {
69  		return sup;
70  	}
71  
72  	public void setExtends(ReferenceType ext) {
73  		this.ext = ext;
74  	}
75  
76  	public void setSuper(ReferenceType sup) {
77  		this.sup = sup;
78  	}
79  
80  }
