192 	 * Explore the CompilationUnit and create appropriate UML elements.
193 	 * This is the main entrance of the analyzer.
194 	 *
195 	 * @param cu The compilation unit to analyze.
196 	 */
197 	public void processCompilationUnit(CompilationUnit cu) {
198 
199 		// First, create the current context.
200 		Package currentCompilationUnitPackage = getCuPackage(cu.getPackage());
201 		LocalContext localContext = new LocalContext(currentCompilationUnitPackage);
202 
203 		classifierCatalog.setCurrentCompilationUnitPackage(currentCompilationUnitPackage);
204 
205 		// Add imports
206 		importedTypes.clear();
207 		if (cu.imports() != null) {
208 			addImports(importedTypes, cu.imports());
209 		}
210 
211 		// Now, create the tree of NamedElement corresponding to declaration.
212 		// Do not set the references to the types
213 		List<AbstractTypeDeclaration> types = cu.types();
214 		for( AbstractTypeDeclaration abstractType : types ) {
215 			Classifier createdElement = processNamedElementDeclaration(abstractType, localContext);
216 		}
217 
218 		// Now, explore the tree of NamedElements and set the types.
219 		for( AbstractTypeDeclaration abstractType : types ) {
220 			processNamedElementReferences(abstractType, localContext);
221 		}
222 
223 	}

226 	 * Process all NamedElement declaration. Walk the tree of declaration in depth first.
227 	 * For each NamedElement declaration, check if it exists, or create it with its name. Do not set or change the references to
228 	 * other types.
229 	 * 
230 	 * @param abstractType
231 	 * @param localContext The current namespace.
232 	 */
233 	private Classifier processNamedElementDeclaration(AbstractTypeDeclaration abstractType, LocalContext localContext) {
234 		switch (abstractType.getNodeType()) {
235 		case ASTNode.TYPE_DECLARATION:
236 			// Class or interface
237 			return processTypeDeclaration((TypeDeclaration)abstractType, localContext);
238 
239 		case ASTNode.ANNOTATION_TYPE_DECLARATION:
240 			return processAnnotationTypeDeclaration((AnnotationTypeDeclaration)abstractType, localContext);
241 
242 		case ASTNode.ENUM_DECLARATION:
243 			return processEnumDeclaration((EnumDeclaration)abstractType, localContext);
244 
245 		default:
246 			return null;
247 		}
248 	}

352 	/**
353 	 * Process TypeDeclaration, which can be Class or Interface.
354 	 * Check if the type already exists in the current namespace. Create it if it does not exist.
355 	 * 
356 	 * @param node The type declaration to found or create
357 	 * @param parentContext The current namespace.
358 	 */
359 	
360 	private Classifier processTypeReferences(TypeDeclaration node, LocalContext parentContext) {
361 
362 		// Create or get this type
363 		Classifier processedClass;
364 		if (node.isInterface()) {
365 			processedClass = lookupInterface(parentContext, node);
366 		} else {
367 			processedClass = lookupClass(parentContext, node);
368 		}
369 
370 		LocalContext localContext = new LocalContext(processedClass, parentContext);
371 
372 		// Process javadoc
373 		processJavadoc(node.getJavadoc(), processedClass);
374 		
375 		// Modifiers
376 		processModifiers(processedClass, node.modifiers());
377 		
378 		// extends and implements
379 		if( node.isInterface()) {
380 			// implements
381 			List<Type> implementedTypes = node.superInterfaceTypes();
382 			for( Type type : implementedTypes) {
383 				Interface generalization = getInterfaceForType( type, localContext);
384 				if (processedClass instanceof BehavioredClassifier) {
385 					UmlUtils.getInterfaceRealization((BehavioredClassifier) processedClass, (Interface) generalization);
386 				}
387 				else {
388 					// should not happen
389 					UmlUtils.getGeneralization(processedClass, generalization);
390 				}
391 			}
392 			
393 		} 
394 		else {
395 			// Class
396 			// Extends parameters
397 			// This only apply to classes
398 			Type superclassType = node.getSuperclassType();
399 			if (superclassType != null) {
400 				Classifier generalization = getClassForType( superclassType, localContext);
401 				// create the generalization
402 				UmlUtils.getGeneralization(processedClass, generalization);
403 			}
404 			// implements
405 			List<Type> implementedTypes = node.superInterfaceTypes();
406 			for( Type type : implementedTypes) {
407 				Classifier generalization = getInterfaceForType( type, localContext);
408 				if (processedClass instanceof BehavioredClassifier) {
409 					UmlUtils.getInterfaceRealization((BehavioredClassifier) processedClass, (Interface) generalization);
410 				}
411 				else {
412 					// should not happen
413 					UmlUtils.getGeneralization(processedClass, generalization);
414 				}
415 			}
416 		}
417 		
418 		
419 		// Now process the properties
420 		for( FieldDeclaration fieldDeclaration : node.getFields()) {
421 			processFieldDeclaration( processedClass, fieldDeclaration, localContext);
422 		}
423 		
424 		// Process methods
425 		for( MethodDeclaration methodDeclaration : node.getMethods()) {
426 			processMethodDeclaration( processedClass, methodDeclaration, localContext);
427 		}
428 		
429 		// Process nested classes
430 		for( TypeDeclaration nestedTypes : node.getTypes()) {
431 			processTypeReferences(nestedTypes, localContext);
432 		}
433 		
434 		return processedClass;
435 	}

859 	/**
860 	 * Get a {@link Classifier} in the UML model that correspond to the specified Type in the reversed class.
861 	 * If the Classifier is not found in the UML model, create it 'a minima' (only the Classifier skeleton).
862 	 * <br>
863 	 * This method is called when a UML Classifier is requested for a type found in a type reference (property, parameters, extends, implements).
864 	 * 
865 	 * @param requestedType The type in java file for which a Classifier in java model is requested
866 	 * @param context The actual context
867 	 * @param type The type of classifier to create. One of {@link UmlUtils#CLASS_TYPE} or {@link UmlUtils#INTERFACE_TYPE}
868 	 * 
869 	 * @return The corresponding Classifier. Should never return null.
870 	 */
871 	private Classifier getClassifierForType(Type requestedType, LocalContext context, EClass type) {
872 
873 		Classifier result;
874 		
875 		// First, extract the name of the requested type
876 		String shortname = JdtAstUtils.getTypeShortname( requestedType );
877 		
878 		
879 		
880 		// Check if the shortname is declared in imports
881 		List<String> qualifiedName = importedTypes.getQualifiedName(shortname);
882 		if( qualifiedName.size() > 1) {
883 			// Found in import, try to get corresponding Classifier, or create it
884 			result =  classifierCatalog.getClassifier(qualifiedName);
885 			if( result != null) {
886 				return result;
887 			}
888 			
889 			// Classifier not found : create it
890 			Package creationPackage = creationPackageCatalog.getCreationPackage(qualifiedName, null);
891 			result = UmlUtils.getClassifier(creationPackage, qualifiedName, type);
892 			return result;
893 		}
894 		
895 		// Check if a corresponding type can be found in context.
896 		result = context.lookupClassifier(shortname, type);
897 		if( result != null) {
898 			return result;
899 		}
900 		
901 		// check in '*' imports
902 		// Only check if the requested classifier exist in UML model.
903 		for( List<String> parentPackageQName : importedTypes.getStarImports() ) {
904 			// Build a possible Fully qualified name
905 			List<String> fullQName = new ArrayList<String>(parentPackageQName);
906 			fullQName.add(shortname);
907 			
908 			result = classifierCatalog.getClassifier(fullQName);
909 			if( result != null) {
910 				return result;
911 			}
912 		}
913 		
914 		// Not found, create it in the current package
915 		Package parentPackage = context.getCurrentPackage();
916 		result = (Classifier) UmlUtils.createType(parentPackage, shortname, type);
917 		return result;
918 	}
