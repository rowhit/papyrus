38  public class ConstraintWithVSLPopupEditorConfiguration extends DefaultXtextDirectEditorConfiguration implements ICustomDirectEditorConfiguration {
39  
40  	public ConstraintWithVSLPopupEditorConfiguration() {
41  		super();
42  	}
43  
44  
45  	@Override
46  	public String getTextToEdit(Object editedObject) {
47  		if (editedObject instanceof Constraint) {
48  			Constraint theConstraint = (Constraint) editedObject;
49  			if (theConstraint.getSpecification() != null && theConstraint.getSpecification() instanceof OpaqueExpression) {
50  				OpaqueExpression spec = (OpaqueExpression) theConstraint.getSpecification();
51  				int indexOfVSLExpression = -1;
52  				int i = 0;
53  				for (String language : spec.getLanguages()) {
54  					if (language.equals("VSL")) {
55  						indexOfVSLExpression = i;
56  					}
57  					i++;
58  				}
59  				return indexOfVSLExpression != -1 ? spec.getBodies().get(indexOfVSLExpression) : "";
60  			} else {
61  				return "";
62  			}
63  		} else {
64  			return "Not a Constraint";
65  		}
66  	}
67  
68  	/**
69  	 * @author CEA LIST
70  	 *
71  	 *         A command for updating the context UML model
72  	 */
73  	protected class UpdateConstraintWithVSLCommand extends AbstractTransactionalCommand {
74  
75  		private Constraint theConstraint;
76  
77  		private String newBody;
78  
79  		/*
80  		 * (non-Javadoc)
81  		 *
82  		 * @see
83  		 * org.eclipse.gmf.runtime.emf.commands.core.command.AbstractTransactionalCommand#doExecuteWithResult(org.eclipse.core.runtime.IProgressMonitor
84  		 * , org.eclipse.core.runtime.IAdaptable)
85  		 */
86  		@Override
87  		protected CommandResult doExecuteWithResult(IProgressMonitor arg0, IAdaptable arg1) throws ExecutionException {
88  
89  			theConstraint.setSpecification(null);
90  
91  			OpaqueExpression opaqueExp = UMLFactory.eINSTANCE.createOpaqueExpression();
92  			opaqueExp.getLanguages().add("VSL");
93  			opaqueExp.getBodies().add(newBody);
94  
95  			theConstraint.setSpecification(opaqueExp);
96  
97  			return CommandResult.newOKCommandResult(theConstraint);
98  		}
99  
100 		public UpdateConstraintWithVSLCommand(Constraint constraint, String newBody) {
101 			super((TransactionalEditingDomain) EMFHelper.resolveEditingDomain(constraint), "Constraint Update", getWorkspaceFiles(constraint));
102 			this.theConstraint = constraint;
103 			this.newBody = newBody;
104 		}
105 	}
106 
107 	@Override
108 	public Injector getInjector() {
109 		return ConstraintWithVSLlActivator.getInstance().getInjector("org.eclipse.papyrus.marte.textedit.constraintwithvsl.xtext.ConstraintWithVSLl");
110 	}
111 
112 
113 	@Override
114 	protected ICommand getParseCommand(EObject umlObject, EObject xtextObject) {
115 		ConstraintSpecification newSpecification = null;
116 
117 		ConstraintRule rule = (ConstraintRule) xtextObject;
118 		newSpecification = rule.getSpecification();
119 		String newBody = null;
120 		if (newSpecification != null) {
121 			newBody = VSLSerializationUtil.printExpression(newSpecification.getExpression());
122 		}
123 
124 		if (!(umlObject instanceof Constraint)) {
125 			return UnexecutableCommand.INSTANCE;
126 		}
127 
128 		if (newBody == null) {
129 			return UnexecutableCommand.INSTANCE;
130 		}
131 
132 		Constraint constraint = (Constraint) umlObject;
133 
134 		UpdateConstraintWithVSLCommand updateCommand = new UpdateConstraintWithVSLCommand(constraint, newBody);
135 
136 		return updateCommand;
137 	}
